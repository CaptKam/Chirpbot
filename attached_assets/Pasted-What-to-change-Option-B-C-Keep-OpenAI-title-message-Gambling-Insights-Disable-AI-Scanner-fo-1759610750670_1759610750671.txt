Safer OpenAI call
private async callOpenAI(payload: { system: string; user: string }) {
  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), AIFeatures.openAITimeoutMs);

  try {
    const res = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type':'application/json',
        'Authorization':`Bearer ${process.env.OPENAI_API_KEY}`
      },
      body: JSON.stringify({
        model: 'gpt-4o-mini',
        temperature: AIFeatures.openAITemperature,
        response_format: { type: 'json_object' },
        messages: [
          { role: 'system', content: payload.system },
          { role: 'user', content: payload.user }
        ],
        max_tokens: 120
      }),
      signal: controller.signal
    });
    const json = await res.json();
    // basic guardrail
    const text = json?.choices?.[0]?.message?.content || '{}';
    const parsed = JSON.parse(text);
    return {
      primary: String(parsed.primary || '').trim(),
      secondary: String(parsed.secondary || '').trim()
    };
  } catch (e) {
    return { primary: '', secondary: '' }; // fall back; don’t block
  } finally {
    clearTimeout(timer);
  }
}

3) Automatic duplicate hider (server-side)
// server/services/text-utils.ts
export function normalize(s: string) {
  return s.toLowerCase().replace(/\s+/g, ' ').replace(/[^\w\s]/g, '').trim();
}
export function jaccardSimilarity(a: string, b: string) {
  const A = new Set(normalize(a).split(' '));
  const B = new Set(normalize(b).split(' '));
  if (!A.size || !B.size) return 0;
  const inter = [...A].filter(x => B.has(x)).length;
  const uni = new Set([...A, ...B]).size;
  return inter / uni;
}


Hook into the combine step:

// server/services/unified-ai-processor.ts (combine)
import { jaccardSimilarity } from './text-utils';
import { AIFeatures } from '../config/ai-features';

function applyDuplicateSuppression(enhanced: UnifiedAIResponse) {
  if (!AIFeatures.hideDuplicateInsights) return enhanced;

  const main = `${enhanced.ai?.primary ?? ''} ${enhanced.ai?.secondary ?? ''}`.trim();
  const keyInsights = (enhanced.gambling?.bulletPoints ?? []).join(' ');
  const sim = jaccardSimilarity(main, keyInsights);

  if (sim >= AIFeatures.duplicateSimilarityThreshold) {
    // hide Key Insights list; keep confidence/tags/markets
    return {
      ...enhanced,
      gambling: {
        ...enhanced.gambling,
        bulletPoints: [],                // ← suppress list
        duplicateSuppressed: true
      }
    };
  }
  return enhanced;
}
