Data model (Prisma)
enum Role { ADMIN OPERATOR VIEWER }

model User {
  id        String   @id @default(cuid())
  email     String   @unique
  role      Role     @default(OPERATOR)
  createdAt DateTime @default(now())

  permissions UserAlertPermission[]
}

model AlertType {
  id          String   @id @default(cuid())
  key         String   @unique   // e.g., "RISP", "RE24_L1", "WEATHER_SHIFT"
  sport       String              // MLB, NFL, NCAAF, etc.
  description String
  enabledDefault Boolean @default(true)
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt
}

model UserAlertPermission {
  id          String   @id @default(cuid())
  userId      String
  alertTypeId String
  allowed     Boolean  // true -> user can use this alert type

  user        User      @relation(fields: [userId], references: [id])
  alertType   AlertType @relation(fields: [alertTypeId], references: [id])

  @@unique([userId, alertTypeId])
}

model Alert {
  id        String   @id @default(cuid())
  gameId    String
  sport     String
  typeKey   String   // FK-like to AlertType.key (denormalized for speed)
  status    String   @default("OPEN") // OPEN, MUTED, ACKED
  payload   Json
  createdAt DateTime @default(now())
}

model GameState {
  id         String   @id @default(cuid())
  sport      String   // MLB, NFL, NCAAF...
  extGameId  String   // external game id from feed
  status     String   // SCHEDULED, LIVE, FINAL
  home       String
  away       String
  clock      String?  // "Q3 05:13" or "Top 7th, 1 out"
  score      String?  // "H 24 - 21 A" or "4-3"
  startUtc   DateTime?
  payload    Json     // raw merged snapshot
  updatedAt  DateTime @updatedAt

  @@index([sport, status])
  @@unique([sport, extGameId])
}

Permission evaluation (simple rule)
isAllowed(user, alertType):
  if exists UserAlertPermission(user, alertType): return allowed
  else return AlertType.enabledDefault


This lets you flip defaults globally, then override per user.

Minimal server glue

Socket server (Node / Next custom server)

// server/socket-admin.ts
import { Server } from "socket.io";
import type { NextApiResponseServerIO } from "./types";

export function initAdminSocket(io: Server) {
  const nsp = io.of("/admin");
  nsp.use(authMiddleware); // check session + role
  nsp.on("connection", (sock) => {
    // Optional: room joins by sport
    sock.on("join", ({ sports }: { sports: string[] }) => {
      sports.forEach(s => sock.join(`sport:${s}`));
    });
  });
}

export function emitGameUpdate(io: Server, game) {
  io.of("/admin").to(`sport:${game.sport}`).emit("game.update", game);
}

export function emitAlert(io: Server, alert) {
  io.of("/admin").to(`sport:${alert.sport}`).emit("alert.new", alert);
}


Permissions API (Next.js route handler)

// app/api/admin/users/[id]/permissions/route.ts
import { prisma } from "@/lib/db";
import { z } from "zod";
import { requireRole } from "@/lib/rbac";

const Body = z.array(z.object({
  alertTypeId: z.string(),
  allowed: z.boolean()
}));

export async function GET(_: Request, { params }: { params: { id: string }}) {
  await requireRole("OPERATOR");
  const rows = await prisma.userAlertPermission.findMany({
    where: { userId: params.id },
    include: { alertType: true }
  });
  return Response.json(rows);
}

export async function PUT(req: Request, { params }: { params: { id: string }}) {
  await requireRole("ADMIN");
  const body = Body.parse(await req.json());
  await prisma.$transaction(
    body.map(p =>
      prisma.userAlertPermission.upsert({
        where: { userId_alertTypeId: { userId: params.id, alertTypeId: p.alertTypeId } },
        update: { allowed: p.allowed },
        create: { userId: params.id, alertTypeId: p.alertTypeId, allowed: p.allowed }
      })
    )
  );
  return Response.json({ ok: true });
}


Effective permission helper (server)

export async function isAlertAllowed(userId: string, typeKey: string) {
  const type = await prisma.alertType.findUnique({ where: { key: typeKey }});
  if (!type) return false;
  const override = await prisma.userAlertPermission.findUnique({
    where: { userId_alertTypeId: { userId, alertTypeId: type.id } }
  });
  return override ? override.allowed : type.enabledDefault;
}

Live Sports page (client stub)
// app/live/page.tsx
"use client";
import { useEffect, useState } from "react";
import { io } from "socket.io-client";

type Game = { sport:string; extGameId:string; status:string; home:string; away:string; score?:string; clock?:string; payload:any };

export default function Live() {
  const [games, setGames] = useState<Record<string, Game>>({});

  useEffect(() => {
    const socket = io(process.env.NEXT_PUBLIC_API_WS + "/admin", { withCredentials: true });
    socket.emit("join", { sports: ["MLB","NFL","NCAAF"] });
    socket.on("game.update", (g: Game) => {
      setGames(prev => ({ ...prev, [g.sport + ":" + g.extGameId]: g }));
    });
    return () => socket.close();
  }, []);

  const rows = Object.values(games);
  return (
    <div className="p-6">
      <h1 className="text-xl font-semibold mb-4">Live Sports Feed</h1>
      <div className="grid grid-cols-1 gap-2">
        {rows.map(g => (
          <div key={g.sport + g.extGameId} className="border rounded-xl p-3">
            <div className="flex justify-between">
              <div className="font-medium">{g.sport} • {g.status}</div>
              <div className="text-sm">{g.clock || "-"}</div>
            </div>
            <div className="mt-1">{g.home} @ {g.away}</div>
            <div className="text-sm opacity-80">{g.score || "-"}</div>
          </div>
        ))}
      </div>
    </div>
  );
}
