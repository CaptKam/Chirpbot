1. POWER_BATTER_ON_DECK

Triggers when a known slugger (high HR / OPS) is on deck and there are runners aboard.

Lets you know a big swing could clear the bases.

private async generatePowerBatterOnDeck(game: any, liveData: any) {
  const offense = liveData?.linescore?.offense ?? {};
  const hasRunner = Boolean(offense.first || offense.second || offense.third);
  if (!hasRunner) return;

  const matchup = liveData?.plays?.currentPlay?.matchup;
  const onDeck = matchup?.onDeck; // MLB feed has onDeck info
  if (!onDeck) return;

  // Simplified check: OPS ≥ .800 or HR ≥ 20 (you can tune)
  const stats = onDeck?.stats?.batting;
  if (stats?.ops && stats.ops >= 0.8 || stats?.homeRuns >= 20) {
    const inning = liveData?.linescore?.currentInning ?? 0;
    const tb = liveData?.linescore?.isTopInning ? 'T' : 'B';

    const alertKey = `${game.gameId}_POWER_BATTER_ON_DECK_${inning}_${tb}`;
    if (!await this.allowAlert(alertKey)) return;

    const score = 85; // baseline
    const message = `💥 POWER BATTER ON DECK: ${onDeck.fullName} with runners on for ${game.awayTeam} @ ${game.homeTeam}`;
    await this.saveRealTimeAlert(alertKey, 'POWER_BATTER_ON_DECK', game.gameId, message, { inning, tb, batter: onDeck.fullName }, score);
  }
}

2. CLUTCH_BATTER_ON_DECK

Similar to above, but looks for high RBI hitter in late innings with tying/go-ahead runs on base.

Adds more “game context.”

private async generateClutchBatterOnDeck(game: any, liveData: any) {
  const inning = liveData?.linescore?.currentInning ?? 0;
  if (inning < 7) return;

  const scoreDiff = Math.abs(game.homeScore - game.awayScore);
  if (scoreDiff > 2) return;

  const offense = liveData?.linescore?.offense ?? {};
  const isRISP = Boolean(offense.second || offense.third);
  if (!isRISP) return;

  const matchup = liveData?.plays?.currentPlay?.matchup;
  const onDeck = matchup?.onDeck;
  if (!onDeck) return;

  const stats = onDeck?.stats?.batting;
  if (stats?.rbi && stats.rbi >= 60) { // or use clutch% if available
    const tb = liveData?.linescore?.isTopInning ? 'T' : 'B';
    const alertKey = `${game.gameId}_CLUTCH_BATTER_ON_DECK_${inning}_${tb}`;
    if (!await this.allowAlert(alertKey)) return;

    const score = 90;
    const message = `🔥 CLUTCH BATTER ON DECK: ${onDeck.fullName} (RBI leader) — ${game.awayTeam} @ ${game.homeTeam}`;
    await this.saveRealTimeAlert(alertKey, 'CLUTCH_BATTER_ON_DECK', game.gameId, message, { inning, tb, batter: onDeck.fullName }, score);
  }
}

await this.generatePowerBatterOnDeck(game, liveData);
await this.generateClutchBatterOnDeck(game, liveData);
