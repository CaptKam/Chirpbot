2) Provider client (OpenWeatherMap)
// server/services/weather/providers/openweather.ts
import fetch from "node-fetch";

export type OWMWeather = {
  windMph: number;             // mph
  windDirToDeg: number | null; // 0–360, direction TOWARD which wind blows
  temperatureF: number | null;
  pressureHpa: number | null;
};

const API = "https://api.openweathermap.org/data/2.5/weather";

export async function fetchOWM(lat: number, lon: number, apiKey: string): Promise<OWMWeather> {
  const url = `${API}?lat=${lat}&lon=${lon}&appid=${apiKey}&units=imperial`;
  const res = await fetch(url, { timeout: 5000 });
  if (!res.ok) throw new Error(`OWM ${res.status}`);
  const j = await res.json();

  // OWM wind.deg = direction the wind is COMING FROM (meteorological)
  // Convert to "to" direction by adding 180°
  const fromDeg = typeof j?.wind?.deg === "number" ? j.wind.deg : null;
  const toDeg = fromDeg == null ? null : ((fromDeg + 180) % 360);

  return {
    windMph: Number(j?.wind?.speed ?? 0), // already mph due to units=imperial
    windDirToDeg: toDeg,
    temperatureF: typeof j?.main?.temp === "number" ? j.main.temp : null,
    pressureHpa: typeof j?.main?.pressure === "number" ? j.main.pressure : null,
  };
}

3) Weather aggregator (+CF vector, caching, domes)
// server/services/enhanced-weather.ts
import { STADIUMS } from "./weather/stadiums";
import { fetchOWM } from "./weather/providers/openweather";

export type EnhancedWeather = {
  windMph: number;
  windDirToDeg: number | null;  // 0..360 (toward)
  cfAzimuthDeg: number | null;
  carryMult: number;            // 1 + 0.018 * outMph
  outMph: number;               // projected mph toward CF (>=0)
  temperatureF?: number | null;
  pressureHpa?: number | null;
  roof?: "open"|"closed"|"retractable";
  tag?: string;                 // e.g., "Wind +18%"
};

const cache = new Map<string, { data: EnhancedWeather; ts: number }>();
const TTL_MS = 90_000;

function carryMultiplier(cfAzimuthDeg: number | null, windDirToDeg: number | null, windMph: number): { outMph: number; mult: number } {
  if (!cfAzimuthDeg && cfAzimuthDeg !== 0) return { outMph: 0, mult: 1 };
  if (windDirToDeg == null || windMph <= 0) return { outMph: 0, mult: 1 };
  let diff = Math.abs(windDirToDeg - cfAzimuthDeg) % 360;
  if (diff > 180) diff = 360 - diff;
  const towardCF = Math.cos((diff * Math.PI) / 180); // +1 out, -1 in
  const outMph = Math.max(0, windMph * towardCF);
  const mult = 1 + 0.018 * outMph;                   // match your V1 scale
  return { outMph, mult };
}

export async function getEnhancedWeather(venueKey: string): Promise<EnhancedWeather> {
  const now = Date.now();
  const cached = cache.get(venueKey);
  if (cached && now - cached.ts < TTL_MS) return cached.data;

  const stadium = STADIUMS[venueKey];
  if (!stadium) {
    const data: EnhancedWeather = { windMph: 0, windDirToDeg: null, cfAzimuthDeg: null, carryMult: 1, outMph: 0 };
    cache.set(venueKey, { data, ts: now }); return data;
  }

  // Roof handling: if closed, zero out carry and suppress tag
  const roof = stadium.roof ?? "open";
  if (roof === "closed") {
    const data: EnhancedWeather = { windMph: 0, windDirToDeg: null, cfAzimuthDeg: stadium.cfAzimuthDeg, carryMult: 1, outMph: 0, roof };
    cache.set(venueKey, { data, ts: now }); return data;
  }

  // Fetch provider
  const apiKey = process.env.OPENWEATHERMAP_API_KEY;
  if (!apiKey) {
    // keep existing conservative behavior if no key (but now explicit)
    const data: EnhancedWeather = { windMph: 0, windDirToDeg: null, cfAzimuthDeg: stadium.cfAzimuthDeg, carryMult: 1, outMph: 0, roof, tag: undefined };
    cache.set(venueKey, { data, ts: now }); return data;
  }

  const owm = await fetchOWM(stadium.lat, stadium.lon, apiKey);
  const { outMph, mult } = carryMultiplier(stadium.cfAzimuthDeg, owm.windDirToDeg, owm.windMph);

  const data: EnhancedWeather = {
    windMph: owm.windMph,
    windDirToDeg: owm.windDirToDeg,
    cfAzimuthDeg: stadium.cfAzimuthDeg,
    carryMult: mult,
    outMph,
    temperatureF: owm.temperatureF,
    pressureHpa: owm.pressureHpa,
    roof,
    tag: outMph >= 5 ? `Wind +${Math.round((mult - 1) * 100)}%` : undefined
  };

  cache.set(venueKey, { data, ts: now });
  return data;
}

6) Rate limits & failures (safe behavior)

Cache per venue for 90s (already in aggregator).

On provider error → carryMult=1 and no wind tag (don’t mislead).

For retractable roofs: if your feed says “closed” tonight, set roof='closed' in runtime and you’ll automatically silence wind.
