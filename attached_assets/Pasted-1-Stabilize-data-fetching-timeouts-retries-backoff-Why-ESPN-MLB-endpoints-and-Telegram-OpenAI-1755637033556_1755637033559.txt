1) Stabilize data fetching (timeouts, retries, backoff)

Why: ESPN/MLB endpoints and Telegram/OpenAI can hiccup; a single slow call can stall loops and spam logs.

Add a tiny fetch helper (server‑wide):

// server/services/http.ts
import pRetry from "p-retry";

export async function fetchJson<T>(
  url: string,
  init: RequestInit & { timeoutMs?: number } = {}
): Promise<T> {
  const { timeoutMs = 5000, ...rest } = init;
  const controller = new AbortController();
  const id = setTimeout(() => controller.abort(), timeoutMs);

  try {
    return await pRetry(async () => {
      const res = await fetch(url, { ...rest, signal: controller.signal });
      if (!res.ok) {
        const body = await res.text().catch(() => "");
        throw new Error(`HTTP ${res.status} ${res.statusText}: ${body.slice(0,200)}`);
      }
      return res.json() as Promise<T>;
    }, { retries: 3, factor: 2, minTimeout: 300, maxTimeout: 2000, randomize: true });
  } finally {
    clearTimeout(id);
  }
}


Then swap your fetch(...) in mlb-api.ts, live-sports.ts, etc. to fetchJson.

2) Prevent alert spam + memory growth

You already hash game state to dedupe, but the map can grow forever and can still jitter on tiny irrelevancies.

Cap and age‑out dedupe keys:

// server/services/engines/base-engine.ts
private lastAlertStates = new Map<string, {hash: string, ts: number}>();
private MAX_KEYS = 5000;
private MAX_AGE_MS = 30 * 60 * 1000; // 30m

protected shouldTriggerAlert(alertType: string, gameId: string, gameState: any): boolean {
  const key = `${gameId}-${alertType}`;
  const stateHash = this.createGameStateHash(gameState, alertType);
  const now = Date.now();

  const prev = this.lastAlertStates.get(key);
  if (prev && prev.hash === stateHash) {
    return false;
  }
  this.lastAlertStates.set(key, { hash: stateHash, ts: now });

  // prune occasionally
  if (this.lastAlertStates.size > this.MAX_KEYS) {
    const cutoff = now - this.MAX_AGE_MS;
    for (const [k, v] of this.lastAlertStates) {
      if (v.ts < cutoff) this.lastAlertStates.delete(k);
    }
  }
  return true;
}


Make the hash ignore noise (e.g., pitch count when the alert is about RISP). In createGameStateHash, pick fields per alert type.

4) WebSocket hardening (keepalive + backpressure)

Add heartbeat to avoid zombie connections, and don’t broadcast to slow clients.

// server/routes.ts (after wss created)
function heartbeat(this: any){ this.isAlive = true; }

wss.on('connection', (ws: any) => {
  ws.isAlive = true;
  ws.on('pong', heartbeat);
});

const interval = setInterval(() => {
  wss.clients.forEach((ws: any) => {
    if (ws.isAlive === false) return ws.terminate();
    ws.isAlive = false;
    ws.ping();
  });
}, 30000);

// make sure to clear on server close if needed


And in broadcast, skip clients with large bufferedAmount:

function broadcast(data: any) {
  const payload = JSON.stringify(data);
  clients.forEach((client) => {
    if (client.readyState === WebSocket.OPEN && client.bufferedAmount < 1_000_000) {
      client.send(payload);
    }
  });
}
