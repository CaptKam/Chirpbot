What to add for SwipeableCard

Three-panels via dragX

Right swipe (delete): reveals red panel; confirm on release beyond threshold.

Center: the alert itself (your photo layout).

Left swipe (AI/Betting): reveals AI panel with books + insight.

Priority dot + badges top-left of the card.

Situation Grid (5 cols): inning/outs/count/bases/priority.

Autoreturn after 3s if user doesn’t complete a swipe.

Dedup affordance: show small “Saved” tick if this alertId is already acknowledged.

Data contract (suggested)
type InningHalf = "Top" | "Bot";
type BaseState = { first: boolean; second: boolean; third: boolean };

interface TeamSide { id: string; name: string; logoUrl: string; score: number; }
interface GameState {
  sport: "MLB" | "NCAAF" | "NBA" | "NHL";
  inning?: number; half?: InningHalf; outs?: number;
  balls?: number; strikes?: number;
  bases?: BaseState;
  countStr?: string; // "1-2"
  clockStr?: string; // for other sports
  atBat?: string; pitcher?: string;
  scheduledAtIso?: string; // if not live
  situationText: string;   // play-by-play line
}

interface Alert {
  id: string;
  type: "RISP" | "HIGH_SCORING" | "CLOSE_GAME" | "NEWS" | string;
  priority: number; // 0..100 (color ramp)
  confidence?: number; // %
  ts: string; // ISO
  away: TeamSide; home: TeamSide;
  game: GameState;
  ai?: { headline?: string; rec?: string; roi?: number };
  seen?: boolean;
}

Motion/gesture spec

Drag axis: x only

Constraints: left −420px, right +160px

Thresholds: complete left ≤ −120 OR velocity < −300px/s; complete right ≥ +90

Elastic: 0.15

Spring: { stiffness: 260, damping: 24, mass: 0.6 }

Auto-return: 3000ms (cancel on pointerdown)

Color/priority ramp

priority 0–39 = gray border

40–69 = amber border/pulse

70–100 = red border/pulse

Accessible bits

role="button" on card; aria-label with game + situation

Keyboard: ←/→ to open panels; Delete to confirm delete; Enter to open detail

Drop-in skeleton (TypeScript + Tailwind + Framer Motion + shadcn/ui)
import * as React from "react";
import { motion, useMotionValue, useTransform, PanInfo } from "framer-motion";
import { Trash2, BrainCircuit, ExternalLink, Clock } from "lucide-react";
import { Card, CardContent } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";

type Props = {
  alert: Alert;
  onOpen?: (id: string) => void;
  onDelete?: (id: string) => Promise<void> | void;
  onAck?: (id: string) => void; // mark seen
  sportsbooks?: Sportsbook[];
  className?: string;
};

export function SwipeableCard({
  alert,
  onOpen,
  onDelete,
  onAck,
  sportsbooks = [],
  className
}: Props) {
  const dragX = useMotionValue(0);
  const [isDragging, setDragging] = React.useState(false);
  const [isDeleting, setDeleting] = React.useState(false);
  const autoRef = React.useRef<number | null>(null);

  // Panel reveals
  const showLeft = useTransform(dragX, [-420, -120, 0], [1, 1, 0]);
  const showRight = useTransform(dragX, [0, 90, 160], [0, 1, 1]);

  React.useEffect(() => {
    // auto-return after 3s when panels are partially open
    const id = window.setInterval(() => {
      const x = dragX.get();
      if (!isDragging && Math.abs(x) > 10) {
        dragX.stop();
        dragX.set(0);
      }
    }, 3000);
    autoRef.current = id;
    return () => { if (autoRef.current) window.clearInterval(autoRef.current); };
  }, [isDragging]);

  const handleDragEnd = async (_: any, info: PanInfo) => {
    setDragging(false);
    const x = info.offset.x;
    const v = info.velocity.x;

    // complete delete (right)
    if (x > 90 || v > 300) {
      setDeleting(true);
      try { await onDelete?.(alert.id); } finally {
        setDeleting(false);
        dragX.set(0);
      }
      return;
    }
    // open AI panel (left)
    if (x < -120 || v < -300) {
      dragX.set(-320);
      return;
    }
    // return to center
    dragX.set(0);
  };

  return (
    <div className={`relative h-[168px] select-none ${className || ""}`}>
      {/* LEFT: AI / Betting panel */}
      <motion.div
        className="absolute inset-0 rounded-2xl p-4 bg-gradient-to-r from-indigo-700 to-fuchsia-600 text-white"
        style={{ opacity: showLeft }}
      >
        <div className="flex items-center gap-2 mb-2">
          <BrainCircuit className="h-5 w-5" />
          <span className="font-semibold">AI Insight</span>
        </div>
        <div className="text-sm line-clamp-3">
          {alert.ai?.headline || "Analyzing live situation..."}
        </div>
        <div className="mt-3 flex gap-2 overflow-x-auto pb-1">
          {sportsbooks.slice(0,5).map(b => (
            <a key={b.name} href={b.appUrl || b.storeUrl} target="_blank" rel="noreferrer">
              <Button size="sm" variant="secondary" className="gap-2">
                <img src={b.logo} alt="" className="h-4 w-4 rounded" />
                {b.name}
                <ExternalLink className="h-3 w-3" />
              </Button>
            </a>
          ))}
        </div>
      </motion.div>

      {/* RIGHT: Delete panel */}
      <motion.div
        className="absolute inset-0 rounded-2xl p-4 bg-gradient-to-l from-rose-600 to-red-500 text-white flex items-center justify-end pr-6"
        style={{ opacity: showRight }}
      >
        <Button disabled={isDeleting} variant="destructive" className="gap-2">
          <Trash2 className="h-4 w-4" /> {isDeleting ? "Deleting..." : "Delete"}
        </Button>
      </motion.div>

      {/* CENTER: Main card */}
      <motion.div
        role="button"
        aria-label={`${alert.away.name} at ${alert.home.name}, ${alert.game.situationText}`}
        tabIndex={0}
        onKeyDown={(e) => {
          if (e.key === "Enter") onOpen?.(alert.id);
          if (e.key === "Delete") onDelete?.(alert.id);
          if (e.key === "ArrowLeft") dragX.set(-320);
          if (e.key === "ArrowRight") dragX.set(140);
        }}
        drag="x"
        dragConstraints={{ left: -420, right: 160 }}
        dragElastic={0.15}
        style={{ x: dragX }}
        onDragStart={() => setDragging(true)}
        onDragEnd={handleDragEnd}
        className="absolute inset-0"
      >
        <Card className={`h-full rounded-2xl border-2 ${priorityBorder(alert.priority)}`}>
          <CardContent className="h-full p-3">
            {/* Header */}
            <div className="flex items-center justify-between">
              <TeamScore team={alert.away} align="left" />
              <CenterBadge game={alert.game} priority={alert.priority} />
              <TeamScore team={alert.home} align="right" />
            </div>

            {/* Play text */}
            <p className="mt-2 text-sm leading-snug line-clamp-2">
              {alert.game.situationText}
            </p>

            {/* Footer + grid */}
            <div className="mt-3 grid grid-cols-5 gap-2 text-xs">
              <InfoPill label="Inning" value={inningStr(alert.game)} />
              <InfoPill label="Outs" value={String(alert.game.outs ?? "-")}/>
              <InfoPill label="Count" value={alert.game.countStr || "-"} />
              <BaseDiamond bases={alert.game.bases}/>
              <PriorityPill value={alert.priority}/>
            </div>

            <div className="mt-2 flex items-center justify-between text-xs text-muted-foreground">
              <div>
                {alert.game.scheduledAtIso
                  ? (<span className="inline-flex items-center gap-1"><Clock className="h-3 w-3" /> First pitch {timeShort(alert.game.scheduledAtIso)}</span>)
                  : (<span>{alert.game.atBat ? `${alert.game.atBat}` : ""}{alert.game.pitcher ? `  •  ${alert.game.pitcher}` : ""}</span>)
                }
              </div>
              <div className="flex items-center gap-2">
                <Badge variant="outline">{alert.type}</Badge>
                {typeof alert.confidence === "number" && (
                  <Badge>{alert.confidence}%</Badge>
                )}
              </div>
            </div>
          </CardContent>
        </Card>
      </motion.div>
    </div>
  );
}

/* ---------- Subcomponents ---------- */

function TeamScore({ team, align }: { team: TeamSide; align: "left" | "right" }) {
  return (
    <div className={`flex items-center gap-2 ${align === "right" ? "flex-row-reverse" : ""}`}>
      <img src={team.logoUrl} alt="" className="h-6 w-6" />
      <span className="text-lg font-semibold">{team.score}</span>
    </div>
  );
}

function CenterBadge({ game, priority }: { game: GameState; priority: number }) {
  return (
    <div className="flex items-center gap-2">
      <span className={`h-2.5 w-2.5 rounded-full ${priorityDot(priority)} animate-pulse`} />
      <div className="rounded-xl bg-muted px-2 py-1 text-xs font-medium">
        {game.half ? `${game.half} ${game.inning}` : game.clockStr || "—"}
      </div>
    </div>
  );
}

function InfoPill({ label, value }: { label: string; value: string }) {
  return (
    <div className="rounded-lg border px-2 py-1">
      <div className="text-[10px] uppercase tracking-wide text-muted-foreground">{label}</div>
      <div className="font-medium">{value}</div>
    </div>
  );
}

function BaseDiamond({ bases }: { bases?: BaseState }) {
  const b = bases || { first:false, second:false, third:false };
  return (
    <div className="flex items-center justify-center">
      <div className="relative h-8 w-8 rotate-45 border rounded-sm">
        <span className={`absolute left-1/2 top-0 -translate-x-1/2 -translate-y-1/2 h-2 w-2 rounded-full ${b.second ? "bg-blue-600" : "border"}`} />
        <span className={`absolute left-0 top-1/2 -translate-x-1/2 -translate-y-1/2 h-2 w-2 rounded-full ${b.third ? "bg-blue-600" : "border"}`} />
        <span className={`absolute left-1/2 bottom-0 -translate-x-1/2 translate-y-1/2 h-2 w-2 rounded-full ${b.first ? "bg-blue-600" : "border"}`} />
      </div>
    </div>
  );
}

function PriorityPill({ value }: { value: number }) {
  return (
    <div className={`rounded-lg px-2 py-1 text-center font-semibold ${priorityBg(value)}`}>
      {value}
    </div>
  );
}

/* ---------- Helpers ---------- */

function priorityBorder(p: number) {
  if (p >= 70) return "border-red-500";
  if (p >= 40) return "border-amber-500";
  return "border-border";
}
function priorityBg(p: number) {
  if (p >= 70) return "bg-red-100 text-red-700";
  if (p >= 40) return "bg-amber-100 text-amber-800";
  return "bg-muted text-foreground";
}
function priorityDot(p: number) {
  if (p >= 70) return "bg-red-500";
  if (p >= 40) return "bg-amber-500";
  return "bg-gray-400";
}
function inningStr(g: GameState) {
  return g.half && g.inning ? `${g.half} ${g.inning}` : "—";
}
function timeShort(iso: string) {
  const d = new Date(iso);
  return d.toLocaleTimeString([], { hour: "numeric", minute: "2-digit" });
}
