// server/index.ts (top-level additions)
import http from "http";
import { Pool } from "pg";
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

// Acquire a DB advisory lock so only one engine ticks
async function acquireEngineLock(): Promise<boolean> {
  const client = await pool.connect();
  try {
    // choose any app-unique key; keep stable across deploys
    const res = await client.query("SELECT pg_try_advisory_lock($1)", [ 842_240_001 ]);
    return res.rows?.[0]?.pg_try_advisory_lock === true;
  } finally { client.release(); }
}

// Start HTTP + WS on one server, fail fast on port conflicts

2) Robust, shared “live” detection

server/services/multi-source-aggregator.ts (or a small util imported everywhere):

// server/services/status.ts
export function isLiveStatus(s: any): boolean {
  const abs = (s?.abstractGameState ?? "").trim();
  const coded = (s?.codedGameState ?? "").trim();
  const text = (s?.detailedState ?? s?.status ?? "").toLowerCase();
  return abs === "Live" || coded === "I" || text === "live" || text.includes("in progress");
}


Use isLiveStatus() anywhere you filter for live games; don’t rely on a single field.

7) Quick DB checks (use whichever timestamp column you have)

Some builds use timestamp, others created_at. Run both to be safe:

-- alerts by hour (yesterday ET window)
select date_trunc('hour', created_at) h, count(*)
from alerts
where created_at >= (now() at time zone 'america/new_york')::date
  and created_at <  (now() at time zone 'america/new_york')::date + interval '1 day'
order by 1;

select date_trunc('hour', "timestamp") h, count(*)
from alerts
where "timestamp" >= (now() at time zone 'america/new_york')::date
  and "timestamp" <  (now() at time zone 'america/new_york')::date + interval '1 day'
order by 1;

-- most recent 20 alerts
select id, type, title, coalesce(created_at, "timestamp") as ts
from alerts
order by coalesce(created_at, "timestamp") desc
limit 20;
