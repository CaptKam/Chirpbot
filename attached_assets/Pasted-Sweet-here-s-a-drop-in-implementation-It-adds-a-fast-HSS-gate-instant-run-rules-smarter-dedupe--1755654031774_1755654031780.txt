// Lightweight High-Scoring Situation (HSS) gate

export type Hand = 'L'|'R';
export type Tier = 'cold'|'warm'|'hot';

export type Ctx = {
  gamePk: number;
  inning: number;
  half: 'top'|'bot';
  outs: 0|1|2;
  runners: { first: boolean; second: boolean; third: boolean };
  li: number;                   // if you don’t have LI, approximate (see note below)
  lead: number;                 // batting team lead (+ ahead, - behind)
  batter: { id: number; iso: number; hr: number; hand: Hand };
  onDeck: { id: number; iso: number; hr: number; hand: Hand };
  pitcher: { throws: Hand; pitchCount: number; timesThrough: 1|2|3|4 };
  oppBullpenERA: number;        // 3.80 typical; fallback to 4.20 if unknown
  wind: { mph: number; toCF: boolean };
  parkHR: number;               // 100 neutral
  balls?: 0|1|2|3;
  strikes?: 0|1|2;
};

export type HssConfig = {
  debounce_after_event_ms: number;
  duplicate_ttl_sec: number;
  negative_ttl_sec: { cold:number; warm:number; hot:number; critical:number };
  thresholds: { cold:number; warm:number; hot:number };
  wind_run_trigger_mph: number;
};

export const defaultCfg: HssConfig = {
  debounce_after_event_ms: 2000,
  duplicate_ttl_sec: 120,
  negative_ttl_sec: { cold:60, warm:25, hot:10, critical:6 },
  thresholds: { cold:0.70, warm:0.55, hot:0.40 },
  wind_run_trigger_mph: 8,
};

const clamp = (x:number, lo=0, hi=1)=> Math.max(lo, Math.min(hi, x));
const runnersKey = (r: Ctx['runners']) => `${+r.first}${+r.second}${+r.third}`;

export function baseOutScore(r:Ctx['runners'], outs:0|1|2): number {
  const f=r.first,s=r.second,t=r.third;
  if (s && t && outs===0) return 1.00;
  if (s && t && outs===1) return 0.85;
  if (f && s && t && outs===1) return 0.90;
  if (t && outs===1 && !f && !s) return 0.70;
  if (!f && !s && !t && outs===2) return 0.05;

  // light defaults
  let v = 0.20;
  if (t) v = Math.max(v, 0.60 - 0.15*outs);
  if (s) v = Math.max(v, 0.45 - 0.10*outs);
  if (f) v = Math.max(v, 0.25 - 0.10*outs);
  return clamp(v);
}

export function hssScore(ctx: Ctx): number {
  const baseOut = baseOutScore(ctx.runners, ctx.outs);
  const li = clamp((ctx.li ?? 1)/3);                               // if LI missing, assume ~1
  const scoreTight = (Math.abs(ctx.lead)<=1)?1:(Math.abs(ctx.lead)===2?0.7:0.2);
  const batterPower = Math.max(ctx.batter.hr>=25?1:0, clamp((ctx.batter.iso||0)/0.25));
  const onDeckPower = ctx.onDeck.hr>=20?1:clamp((ctx.onDeck.iso||0)/0.22);
  const platoon = (ctx.batter.hand!==ctx.pitcher.throws)?1:0.5;
  const fatigue = clamp((ctx.pitcher.timesThrough>=3?1:0)*0.6 + (ctx.pitcher.pitchCount/100)*0.4);
  const penQual = clamp(1 - (ctx.oppBullpenERA||4.2)/5);
  const windCF = ctx.wind.toCF ? clamp((ctx.wind.mph||0)/12) : 0;
  const park = clamp((ctx.parkHR||100)/110, 0.5, 1);

  return 0.30*baseOut + 0.15*li + 0.10*scoreTight + 0.12*batterPower +
         0.08*onDeckPower + 0.05*platoon + 0.05*fatigue + 0.05*penQual +
         0.07*windCF + 0.03*park;
}

export function tier(ctx: Ctx): Tier {
  const risp = ctx.runners.second || ctx.runners.third;
  const late = (ctx.inning>=7);
  const tight = Math.abs(ctx.lead)<=2;
  if ((late && Math.abs(ctx.lead)<=1) || (ctx.li ?? 1) >= 2) return 'hot';
  if (risp || tight) return 'warm';
  return 'cold';
}

export function instantRun(ctx: Ctx, cfg = defaultCfg): boolean {
  const risp23_1out = ctx.runners.second && ctx.runners.third && ctx.outs===1;
  const powerOnDeck = ctx.onDeck.hr>=20;
  const windHelpful = ctx.wind.toCF && ctx.wind.mph>=cfg.wind_run_trigger_mph;
  if (ctx.runners.first && ctx.runners.second && ctx.runners.third) return true;
  if (risp23_1out && powerOnDeck && windHelpful) return true;
  if ((ctx.inning>=9) && Math.abs(ctx.lead)<=1 && (ctx.runners.second||ctx.runners.third)) return true;
  return false;
}

export function shouldCallLLM(
  ctx: Ctx,
  now: number,
  nextAllowedEvalAt: number,
  thresholds = defaultCfg.thresholds
): boolean {
  if (now < nextAllowedEvalAt) return false;
  if (instantRun(ctx)) return true;
  const sc = hssScore(ctx);
  const t = tier(ctx);
  const th = (t==='hot')?thresholds.hot : t==='warm'?thresholds.warm : thresholds.cold;
  return sc >= th;
}

export function buildDupKey(ctx: Ctx): string {
  const windBucket = `${ctx.wind.toCF?'cf':'other'}:${Math.round((ctx.wind.mph||0)/2)*2}`;
  const countBucket = `${ctx.balls ?? -1}-${ctx.strikes ?? -1}`;
  return [
    ctx.inning, ctx.half, ctx.outs, runnersKey(ctx.runners),
    ctx.batter.id, ctx.onDeck.id, windBucket, countBucket, Math.sign(ctx.lead)
  ].join('|');
}

function ctxFromFeed(feed:any): Ctx {
  // Map MLB feed → Ctx (fill with your actual paths)
  const ls = feed.liveData.linescore;
  const box = feed.liveData.boxscore;
  const plays = feed.liveData.plays;
  const weather = feed.gameData.weather || {};
  const windText = (weather.wind || '').toLowerCase(); // e.g., "10 mph, Out to Center"
  const windMph = Number((windText.match(/(\d+)\s*mph/)?.[1]) || 0);
  const toCF = /center/.test(windText) || /out to\s*center/.test(windText);

  const batter = box?.teams?.[offenseIsHome?'home':'away']?.batters?.[0]; // replace with current batter id lookup
  const onDeck = box?.teams?.[offenseIsHome?'home':'away']?.batters?.[1];

  const pitcher = box?.teams?.[offenseIsHome?'away':'home']?.pitchers?.[0];

  // You likely have these cached; provide safe fallbacks:
  const ctx: Ctx = {
    gamePk: feed.gamePk,

    half: (ls.inningState==='Top'?'top':'bot'),
    outs: (ls.outs ?? 0) as 0|1|2,
    runners: {
      first: !!ls.offense?.first,
      second: !!ls.offense?.second,
      third: !!ls.offense?.third,
    },

    onDeck: { id: onDeck?.person?.id ?? -1, iso: onDeck?.stats?.batting?.iso ?? 0.18, hr: onDeck?.seasonStats?.batting?.homeRuns ?? 10, hand: (onDeck?.hand?.bat || 'R') as any },
    pitcher: { throws: (pitcher?.hand?.throw || 'R') as any, pitchCount: pitcher?.stats?.pitching?.numberOfPitches ?? 40, timesThrough: (pitcher?.stats?.pitching?.battersFaced ?? 9) >= 18 ? 2 : 1 },

async function processGame(feed:any) {
  const ctx = ctxFromFeed(feed);
  const st = gameStateMap.get(ctx.gamePk) ?? {};
  const now = Date.now();

  // Debounce: skip bursts (e.g., multiple partial updates inside 2s)
  if (st.lastEventAt && (now - st.lastEventAt) < HSS.debounce_after_event_ms) return;

  const dupKey = buildDupKey(ctx);
  if (st.lastDupKey === dupKey && st.lastDupAt && (now - st.lastDupAt) < HSS.duplicate_ttl_sec*1000) {
    st.lastEventAt = now;
    gameStateMap.set(ctx.gamePk, st);
    return;
  }

  // Budget guard (per-minute)
  if (!budgetOK()) return;

  const allowed = FEATURES.llm_hss_only
    ? shouldCallLLM(ctx, now, st.nextAllowedEvalAt ?? 0, HSS.thresholds)
    : true; // legacy path

  if (!allowed) {
    // if not allowed, still update lastEventAt to respect debounce
    st.lastEventAt = now;
    gameStateMap.set(ctx.gamePk, st);
    return;
  }

  // ---- Call your existing OpenAI pipeline only here ----
  const result = await runOpenAIForAlert(ctx); // your function

  // TTL management
  const t = tier(ctx);
  const negTTL = (t==='hot') ? HSS.negative_ttl_sec.hot :
                 (t==='warm')? HSS.negative_ttl_sec.warm :
                               HSS.negative_ttl_sec.cold;

  st.nextAllowedEvalAt = now + (result?.alertFired ? 3000 : negTTL*1000); // shorter TTL if we just fired
  st.lastDupKey = dupKey;
  st.lastDupAt = now;
  st.lastEventAt = now;
  gameStateMap.set(ctx.gamePk, st);
}

// ---- simple per-minute budget ----
const budget = { used: 0, windowStart: Date.now(), maxPerMin: 450 };
function budgetOK(): boolean {
  const now = Date.now();
  if (now - budget.windowStart >= 60_000) {
    budget.windowStart = now; budget.used = 0;
  }
  if (budget.used >= budget.maxPerMin) return false;
  budget.used++;
  return true;
}

Notes / tips

Leverage Index (LI): if you don’t have it, approximate quickly:

li ≈ baseOutMultiplier * scoreTightMultiplier * inningMultiplier, where
baseOutMultiplier = { RISP:1.2, BL:1.6, else:1.0 },
scoreTightMultiplier = { |lead|≤1:1.3, =2:1.1, else:0.9 },
inningMultiplier = { 7+:1.4, 1–6:1.0 }.
Clamp 0.8–3.0.

Instant‑run rule covers your exact ask: 2nd & 3rd, 1 out + on‑deck ≥20 HR + wind to CF ≥8 mph.

Priority thresholding stays as‑is (e.g., Telegram only ≥75). You’ll naturally produce fewer, sharper AI alerts.

Settings: if “Runners on Base” was off, keep it off; the HSS gate uses raw state, not that toggle.
