What this Tennis engine emits

Alert types & priorities (defaults you can tune):

TENNIS_BREAK_POINT — 85

TENNIS_DOUBLE_BREAK_POINT — 92

TENNIS_SET_POINT — 93

TENNIS_MATCH_POINT — 98

TENNIS_TIEBREAK_START — 80

TENNIS_MOMENTUM_SURGE (won 4 of last 5 points) — 78 (silent toast; feed only)

Dedup scopes (to prevent spam):

Point-scoped: BREAK_POINT, DOUBLE_BREAK_POINT, SET_POINT, MATCH_POINT
Key: matchId:type:set:game:serverSide:pointToken • window 10–15s, escalationOnly=true

Game-scoped: TIEBREAK_START (one per tiebreak)
Key: matchId:type:set • window 30s

Rolling-window: MOMENTUM_SURGE (per set)
Key: matchId:type:set:side • window 60s, realertAfterMs=180s

1) Source interface (plug any tennis feed)

Create server/services/sources/tennis-sources.ts:

// server/services/sources/tennis-sources.ts
export type Side = "home" | "away";

export type TennisMatch = {
  matchId: string;
  tournament?: string;
  round?: string;
  status: { // normalized
    abstract: "Preview" | "Live" | "Final" | "Suspended";
    detailed?: string;
  };
  players: {
    home: { name: string; rank?: number };
    away: { name: string; rank?: number };
  };
  server: Side | null;                 // who serves current point
  setsWon: { home: number; away: number };
  currentSet: number;                  // 1-based
  gamesInSet: { home: number; away: number };
  isTiebreak: boolean;
  // Traditional scoring in current game (when not tiebreak)
  points: { home: "0"|"15"|"30"|"40"|"Ad"|"—"; away: "0"|"15"|"30"|"40"|"Ad"|"—" } | null;
  // Tiebreak points (when isTiebreak)
  tbPoints?: { home: number; away: number };
  // For momentum: last point winner side (if your source provides)
  lastPointWinner?: Side | null;
  // Monotonic token to identify the point (e.g., play sequence id)
  pointToken?: string | number;
};

export interface TennisSource {
  name: string;
  enabled: boolean;
  fetchLiveMatches(): Promise<TennisMatch[]>;
}

// Simple aggregator; plug your concrete source(s) here.
export class TennisAggregator {
  constructor(private sources: TennisSource[]) {}
  async getLiveMatches(): Promise<TennisMatch[]> {
    const enabled = this.sources.filter(s => s.enabled);
    const results = await Promise.allSettled(enabled.map(s => s.fetchLiveMatches()));
    const matches: TennisMatch[] = [];
    for (const r of results) if (r.status === "fulfilled") matches.push(...r.value);
    // Deduplicate by matchId, prefer entries with richer fields
    const byId = new Map<string, TennisMatch>();
    for (const m of matches) {
      const prev = byId.get(m.matchId);
      if (!prev || scoreRichness(m) > scoreRichness(prev)) byId.set(m.matchId, m);
    }
    return [...byId.values()].filter(m => m.status.abstract === "Live");
  }
}

function scoreRichness(m: TennisMatch) {
  let s = 0;
  if (m.points) s += 2;
  if (m.tbPoints) s += 2;
  if (m.pointToken != null) s += 1;
  if (m.lastPointWinner) s += 1;
  return s;
}


Add your concrete source (ESPN-style, etc.) by implementing fetchLiveMatches() and pushing it into new TennisAggregator([...]). No fake data; only live fetch.

2) Tennis engine

Create server/services/engines/tennis-engine.ts:

// server/services/engines/tennis-engine.ts
import crypto from "node:crypto";
import { TennisAggregator, TennisMatch, Side } from "../sources/tennis-sources";
import { sendTelegramAlert } from "../notifiers/telegram"; // your existing
import { alertDeduplication } from "../alert-deduplication"; // your existing
import { enhanceHighPriorityAlert } from "../ai-analysis"; // your existing enhancer

type AlertData = {
  id: string;
  type: string;
  title: string;
  description: string;
  priority: number;
  sport: "TENNIS";
  probability?: number; // not used here; keep shape
  game_info: any;
  created_at: string;
};

export class TennisEngine {
  private momentum = new Map<string, Side[]>(); // matchId -> last N point winners
  private armed = false;
  private metrics = { live: 0, attempts: 0, emitted: 0, suppressed: 0, drops: {} as Record<string,number>, ai: 0 };

  constructor(private aggregator: TennisAggregator, private telegramCfg: any) {
    setInterval(() => {
      console.log(JSON.stringify({ t:"METRICS_TENNIS", ...this.metrics }));
    }, 60_000);
  }

  async monitor() {
    const matches = await this.aggregator.getLiveMatches();
    this.metrics.live = matches.length;
    if (!this.armed && matches.length) {
      this.armed = true;
      console.log(JSON.stringify({ t:"ENGINE_ARMED", sport:"TENNIS", liveGames: matches.length }));
    }
    for (const raw of matches) {
      const s = normalize(raw);
      if (!s) continue;
      // Emit state sample for diagnostics on changes (optional: keep last hash per match)
      console.log(JSON.stringify({
        t:"STATE", sport:"TENNIS", matchId: s.matchId, set: s.currentSet,
        games: s.gamesInSet, points: s.isTiebreak ? s.tbPoints : s.points,
        server: s.server, isTB: s.isTiebreak, pointToken: s.pointToken ?? null
      }));

      const alerts = this.checkAlertConditions(s);
      if (alerts.length) await this.processAlerts(alerts, s);
      this.recordMomentum(s);
    }
  }

  private recordMomentum(s: TennisMatch) {
    if (!s.lastPointWinner) return;
    const arr = this.momentum.get(s.matchId) ?? [];
    arr.push(s.lastPointWinner);
    if (arr.length > 10) arr.shift();
    this.momentum.set(s.matchId, arr);
  }

  private checkAlertConditions(s: TennisMatch): AlertData[] {
    const out: AlertData[] = [];
    const ctx = baseInfo(s);

    // 1) Break point / double break
    if (!s.isTiebreak && s.server) {
      const bp = isBreakPoint(s, s.server);
      if (bp) {
        const type = bp.double ? "TENNIS_DOUBLE_BREAK_POINT" : "TENNIS_BREAK_POINT";
        const priority = bp.double ? 92 : 85;
        const key = dedupKey(s, type);
        this.metrics.attempts++;
        console.log(JSON.stringify({ t:"ALERT_ATTEMPT", type, key, prio: priority }));

        if (!alertDeduplication.shouldAllow(type, s.matchId, key, { escalationOnly: true })) {
          this.metrics.suppressed++; 
        } else {
          out.push(makeAlert(type, bp.double ? "Double Break Point" : "Break Point", priority,
            `${bp.sideName} has ${bp.double ? "DOUBLE break point" : "break point"} (${formatPoints(s)}).`,
            ctx));
        }
      }
    }

    // 2) Set point / Match point
    const sp = setOrMatchPoint(s);
    if (sp) {
      const type = sp.kind === "match" ? "TENNIS_MATCH_POINT" : "TENNIS_SET_POINT";
      const priority = sp.kind === "match" ? 98 : 93;
      const key = dedupKey(s, type);
      this.metrics.attempts++;
      console.log(JSON.stringify({ t:"ALERT_ATTEMPT", type, key, prio: priority }));
      if (!alertDeduplication.shouldAllow(type, s.matchId, key, { escalationOnly: true })) {
        this.metrics.suppressed++;
      } else {
        out.push(makeAlert(type, sp.kind === "match" ? "Match Point" : "Set Point", priority,
          `${sp.sideName} has ${sp.kind === "match" ? "MATCH" : "SET"} POINT (${formatPoints(s)}).`,
          ctx));
      }
    }

    // 3) Tiebreak start
    if (tiebreakStart(s)) {
      const type = "TENNIS_TIEBREAK_START";
      const key = `${s.matchId}:${type}:${s.currentSet}`;
      this.metrics.attempts++;
      console.log(JSON.stringify({ t:"ALERT_ATTEMPT", type, key, prio: 80 }));
      if (!alertDeduplication.shouldAllow(type, s.matchId, key)) {
        this.metrics.suppressed++;
      } else {
        out.push(makeAlert(type, "Tiebreak Start", 80,
          `Tiebreak in Set ${s.currentSet} — score ${s.tbPoints?.home ?? 0}-${s.tbPoints?.away ?? 0}.`,
          ctx));
      }
    }

    // 4) Momentum surge (won 4 of last 5 points)
    const surge = this.momentumSurge(s);
    if (surge) {
      const type = "TENNIS_MOMENTUM_SURGE";
      const key = `${s.matchId}:${type}:${s.currentSet}:${surge.side}`;
      this.metrics.attempts++;
      console.log(JSON.stringify({ t:"ALERT_ATTEMPT", type, key, prio: 78 }));
      if (!alertDeduplication.shouldAllow(type, s.matchId, key, { realertAfterMs: 180000 })) {
        this.metrics.suppressed++;
      } else {
        out.push(makeAlert(type, "Momentum Surge", 78,
          `${surge.sideName} has won ${surge.won} of last ${surge.of} points.`,
          ctx));
      }
    }

    return out;
  }

  private momentumSurge(s: TennisMatch) {
    const arr = this.momentum.get(s.matchId);
    if (!arr || arr.length < 5) return null;
    const last5 = arr.slice(-5);
    const home = last5.filter(x => x === "home").length;
    const away = 5 - home;
    if (home >= 4) return { side: "home" as Side, sideName: s.players.home.name, won: home, of: 5 };
    if (away >= 4) return { side: "away" as Side, sideName: s.players.away.name, won: away, of: 5 };
    return null;
  }

  private async processAlerts(alerts: AlertData[], s: TennisMatch) {
    for (const a of alerts) {
      // Optional AI enhancement (decorate only)
      if (a.priority >= 85) {
        try {
          const enhanced = await enhanceHighPriorityAlert(a.type, {
            homeTeam: s.players.home.name, awayTeam: s.players.away.name,
            score: { home: s.gamesInSet.home, away: s.gamesInSet.away }
          } as any, a.description, a.priority);
          if (enhanced?.enhancedDescription) a.description = enhanced.enhancedDescription;
          this.metrics.ai++;
        } catch (e) {
          console.warn("AI_ERR_TENNIS", (e as any)?.message ?? String(e));
        }
      }

      // Fan-out (WS handled by your existing pipeline); Telegram optional
      try {
        await sendTelegramAlert(this.telegramCfg, a);
      } catch (e) {
        console.warn("TEL_ERR_TENNIS", (e as any)?.message ?? String(e));
      }

      // Persist
      // storage.createAlert(a) — use your existing storage layer
      // For logs:
      this.metrics.emitted++;
      console.log(JSON.stringify({ t:"ALERT_EMIT", type: a.type, id: a.id, prio: a.priority }));
    }
  }
}

// ---------- helpers ----------

function normalize(m: TennisMatch): TennisMatch | null {
  if (m.status.abstract !== "Live") return null;
  return m;
}

function baseInfo(s: TennisMatch) {
  return {
    matchId: s.matchId,
    tournament: s.tournament,
    round: s.round,
    players: s.players,
    server: s.server,
    set: s.currentSet,
    setScore: s.setsWon,
    games: s.gamesInSet,
    isTiebreak: s.isTiebreak,
    points: s.points,
    tbPoints: s.tbPoints
  };
}

function formatPoints(s: TennisMatch) {
  if (s.isTiebreak) return `TB ${s.tbPoints?.home ?? 0}-${s.tbPoints?.away ?? 0}`;
  const p = s.points ?? { home:"—", away:"—" };
  return `${p.home}-${p.away}`;
}

function dedupKey(s: TennisMatch, type: string) {
  const pt = s.pointToken ?? `${s.currentSet}:${s.gamesInSet.home}-${s.gamesInSet.away}:${s.points?.home ?? "-"}-${s.points?.away ?? "-"}`;
  return `${s.matchId}:${type}:S${s.currentSet}:G${s.gamesInSet.home}-${s.gamesInSet.away}:SV${s.server ?? "-"}:PT${pt}`;
}

// ---- tennis rules ----

function isBreakPoint(s: TennisMatch, server: Side) {
  if (s.isTiebreak || !s.points) return null;
  const receiver: Side = server === "home" ? "away" : "home";
  const sp = s.points[server], rp = s.points[receiver];
  // Double BP: receiver 0–40 or 15–40 (two chances)
  const double = (rp === "40" && (sp === "0" || sp === "15"));
  // Single BP: receiver 40–30, or Advantage receiver at deuce
  const single = (rp === "40" && sp === "30") || (rp === "Ad");
  if (double || single) {
    const sideName = s.players[receiver].name;
    return { double, side: receiver, sideName };
  }
  return null;
}

function isGamePointFor(side: Side, s: TennisMatch) {
  if (s.isTiebreak) return false;
  if (!s.points) return false;
  const me = s.points[side], opp = s.points[opposite(side)];
  return (me === "40" && (opp === "0" || opp === "15" || opp === "30")) || (me === "Ad");
}

function opposite(side: Side): Side {
  return side === "home" ? "away" : "home";
}

function setOrMatchPoint(s: TennisMatch) {
  const targetSets = 2; // best of 3 by default; if you expose best-of-5, make this dynamic
  if (s.isTiebreak) {
    const tb = s.tbPoints ?? { home: 0, away: 0 };
    const lead = tb.home - tb.away;
    const homePointAway = tb.home >= 6 && lead >= 1;
    const awayPointAway = tb.away >= 6 && -lead >= 1;
    if (homePointAway || awayPointAway) {
      const side: Side = homePointAway ? "home" : "away";
      const setsWon = s.setsWon[side];
      const kind = (setsWon === targetSets - 1) ? "match" : "set";
      return { side, sideName: s.players[side].name, kind } as const;
    }
    return null;
  }

  // Not tiebreak: side is at game point AND winning this game would clinch set
  for (const side of ["home","away"] as Side[]) {
    if (!isGamePointFor(side, s)) continue;
    const myGames = s.gamesInSet[side], oppGames = s.gamesInSet[opposite(side)];
    // If I win this game → myGames+1; set if (myGames+1 >= 6 && (myGames+1 - oppGames) >= 2)
    const next = myGames + 1;
    const willWinSet = next >= 6 && (next - oppGames) >= 2;
    if (willWinSet) {
      const setsWon = s.setsWon[side];
      const kind = (setsWon === targetSets - 1) ? "match" : "set";
      return { side, sideName: s.players[side].name, kind } as const;
    }
  }
  return null;
}

function makeAlert(type: string, title: string, priority: number, description: string, ctx: any): AlertData {
  return {
    id: crypto.randomUUID(),
    type, title, priority, description,
    sport: "TENNIS",
    game_info: ctx,
    created_at: new Date().toISOString()
  };
}

5) Client adapter (sport-agnostic card works)

Create client/src/adapters/tennis.tsx:

import { adapters, SportAdapter } from "./base";
import type { AlertVM } from "@/lib/alert-vm";

function period(set:number, games:{home:number;away:number}) {
  return `Set ${set} • ${games.home}-${games.away}`;
}

function situation(a:any) {
  const g = a.game_info;
  if (a.type === "TENNIS_TIEBREAK_START") return `Tiebreak • Set ${g.set}`;
  if (a.type?.startsWith("TENNIS_MATCH_POINT")) return `Match Point • ${g.players?.home?.name ?? ""} vs ${g.players?.away?.name ?? ""}`;
  if (a.type?.startsWith("TENNIS_SET_POINT")) return `Set Point • ${g.players?.home?.name ?? ""} vs ${g.players?.away?.name ?? ""}`;
  if (a.type?.includes("DOUBLE_BREAK_POINT")) return `Double Break Point`;
  if (a.type?.includes("BREAK_POINT")) return `Break Point`;
  if (a.type?.includes("MOMENTUM_SURGE")) return `Momentum Surge`;
  return a.title || a.type?.replace(/_/g," ");
}

const tennis: SportAdapter = {
  sport: "TENNIS",
  canHandle: a => a.sport === "TENNIS",
  toViewModel: (a:any): AlertVM => {
    const g = a.game_info || {};
    const scoreline = `${g.players?.home?.name ?? "Home"} — ${g.players?.away?.name ?? "Away"}`;
    const edgeLabel = a.type === "TENNIS_MOMENTUM_SURGE" ? "Run" :
                      a.type?.includes("BREAK_POINT") ? "BP" :
                      a.type?.includes("TIEBREAK") ? "TB" :
                      a.type?.includes("SET_POINT") ? "SP" :
                      a.type?.includes("MATCH_POINT") ? "MP" : "Edge";
    const edgeValue = a.type?.includes("MOMENTUM_SURGE") ? (a.description.match(/\d of last \d/)?.[0] ?? "") : (g.points ? `${g.points.home}-${g.points.away}` : "");
    return {
      id: a.id,
      sport: "TENNIS",
      title: situation(a),
      situation: g.server ? `Serve: ${g.server === "home" ? g.players?.home?.name : g.players?.away?.name}` : "Serve: —",
      scoreline,
      period: period(g.set, g.games ?? {home:0,away:0}),
      edge: { label: edgeLabel, value: edgeValue },
      priority: a.priority ?? 80,
      actionLine: a.description,
      tags: [a.type?.includes("MATCH_POINT") ? "Match Point" : null, g.isTiebreak ? "Tiebreak" : null].filter(Boolean) as string[],
      isNew: !a.seen,
      createdAt: a.created_at
    };
  }
};
