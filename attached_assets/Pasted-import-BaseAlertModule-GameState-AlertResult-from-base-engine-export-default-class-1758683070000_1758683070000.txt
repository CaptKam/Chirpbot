import { BaseAlertModule, GameState, AlertResult } from '../../base-engine';

export default class HighScoringQuarterModule extends BaseAlertModule {
  alertType = 'WNBA_HIGH_SCORING_QUARTER';
  sport = 'WNBA';

  private readonly Q_SEC = 600; // 10:00
  private readonly OT_SEC = 300; // 5:00
  private readonly MIN_ELAPSED_FOR_PACE = 180; // need 3+ min before trusting pace

  // Track per-game per-quarter alert to prevent duplicates: `${gameId}:Q${q}`
  private fired = new Set<string>();

  // Baseline totals at the START of the current quarter (used if feed lacks quarterPoints)
  private baselines = new Map<string, { quarter: number; home: number; away: number }>();

  // ---------- Utils ----------
  private parseClockToSeconds(clock?: string): number | null {
    if (!clock) return null;
    const token = clock.trim().split(' ')[0];
    const [m, s] = token.split(':').map(x => parseInt(x, 10));
    if (Number.isNaN(m) || Number.isNaN(s)) return null;
    return m * 60 + s;
  }

  private key(gs: GameState): string {
    return `${gs.gameId}:Q${gs.quarter ?? 0}`;
  }

  private ensureBaseline(gs: GameState): void {
    const gameId = gs.gameId;
    const q = gs.quarter ?? 0;
    if (!gameId || !q) return;

    const existing = this.baselines.get(gameId);
    if (!existing || existing.quarter !== q) {
      const home = gs.homeScore ?? 0;
      const away = gs.awayScore ?? 0;
      this.baselines.set(gameId, { quarter: q, home, away });
    }
  }

  private quarterPoints(gs: GameState): number | null {
    const q = gs.quarter ?? 0;

    // Prefer explicit quarter scoring if your feed provides it
    // e.g., gs.quarterPoints (combined), or gs.periodScoring[q] = {home, away}
    if (typeof (gs as any).quarterPoints === 'number') {
      return (gs as any).quarterPoints;
    }
    const ps = (gs as any).periodScoring;
    if (ps && ps[q] && typeof ps[q].home === 'number' && typeof ps[q].away === 'number') {
      return ps[q].home + ps[q].away;
    }

    // Fallback: compute from baseline totals
    const base = this.baselines.get(gs.gameId);
    if (base && base.quarter === q) {
      const home = gs.homeScore ?? 0;
      const away = gs.awayScore ?? 0;
      return Math.max(0, (home - base.home) + (away - base.away));
    }

    return null;
  }

  private projection(gs: GameState, qPts: number | null): number | null {
    if (qPts == null) return null;
    const q = gs.quarter ?? 0;
    const secRem = this.parseClockToSeconds(gs.timeRemaining);
    if (secRem == null) return null;

    const totalSec = q >= 5 ? this.OT_SEC : this.Q_SEC;
    const elapsed = totalSec - secRem;
    if (elapsed <= this.MIN_ELAPSED_FOR_PACE) return null; // too early to trust

    const pace = (qPts / elapsed) * totalSec;
    return pace;
  }

  private triggeredAlready(gs: GameState): boolean {
    return this.fired.has(this.key(gs));
  }

  private markFired(gs: GameState): void {
    this.fired.add(this.key(gs));
  }

  // ---------- Core ----------
  isTriggered(gs: GameState): boolean {
    if (!gs?.gameId || !(gs.quarter >= 1)) return false;
    if (!gs.isLive && (gs.status || '').toLowerCase() !== 'live') return false;

    // On quarter change, set the baseline so we can compute quarter points
    this.ensureBaseline(gs);

    // Only trigger once per quarter
    if (this.triggeredAlready(gs)) return false;

    const q = gs.quarter!;
    const qPts = this.quarterPoints(gs);
    const proj = this.projection(gs, qPts);

    // Thresholds
    const isOT = q >= 5;
    const hardNow = isOT ? 28 : 46;     // "already scored" threshold
    const hardProj = isOT ? 30 : 50;    // "on pace for" threshold

    const nowHit = qPts != null && qPts >= hardNow;
    const projHit = proj != null && proj >= hardProj;

    if (nowHit || projHit) {
      this.markFired(gs);
      return true;
    }
    return false;
  }

  generateAlert(gs: GameState): AlertResult | null {
    const q = gs.quarter ?? 0;
    const qPts = this.quarterPoints(gs) ?? 0;
    const secRem = this.parseClockToSeconds(gs.timeRemaining);
    const totalSec = q >= 5 ? this.OT_SEC : this.Q_SEC;
    const proj = this.projection(gs, qPts);
    const paceTxt = proj ? `pace ${Math.round(proj)} pts` : undefined;

    const timeTxt = secRem != null ? `${Math.floor(secRem / 60)}:${(secRem % 60).toString().padStart(2,'0')} left` : 'time N/A';
    const label = q >= 5 ? `OT high-scoring` : `High-scoring Q${q}`;

    const alertKey = `${gs.gameId}_high_scoring_Q${q}`;
    const msgParts = [`🏀 ${label}`, `${qPts} pts`, timeTxt, paceTxt].filter(Boolean);
    const message = msgParts.join(' — ');

    // Priority scales with points and how early it is (earlier + high pace = higher)
    let priority = 78;
    priority += Math.min(12, Math.max(0, qPts - 40)); // boost for big quarters
    if (proj && proj >= (q >= 5 ? 32 : 52)) priority += 4;

    return {
      alertKey,
      type: this.alertType,
      message,
      context: {
        gameId: gs.gameId,
        sport: this.sport,
        homeTeam: gs.homeTeam,
        awayTeam: gs.awayTeam,
        homeScore: gs.homeScore ?? 0,
        awayScore: gs.awayScore ?? 0,
        quarter: q,
        timeRemaining: gs.timeRemaining,
        quarterPoints: qPts,
        projectedQuarterPoints: proj ? Math.round(proj) : undefined,
      },
      priority: Math.min(priority, 95),
    };
  }
