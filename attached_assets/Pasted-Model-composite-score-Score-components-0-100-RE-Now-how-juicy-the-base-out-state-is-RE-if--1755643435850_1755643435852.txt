Model: composite score

Score components (0–100):

RE Now (how juicy the base/out state is)

ΔRE if ball in play (expected change on typical outcomes)

GameContext (inning + closeness)

BatterThreat (current batter)

OnDeckThreat (e.g., Ohtani)

WeatherBoost (tailwind toward outfield)

ParkFactor (HR or run factor)

PitcherState (fatigue / handedness mismatch) – optional

Weighted sum → AlertScore, then map to your existing Priority.

// server/services/engines/mlb-scorer.ts
// Lightweight, deterministic, and fast.

type BaseState = '---' | '1--' | '-2-' | '--3' | '12-' | '1-3' | '-23' | '123';

const RUN_EXPECTANCY: Record<number, Record<BaseState, number>> = {
  // Approx modern MLB RE to end of inning (coarse but useful). Units: runs.
  0: { '---': 0.50, '1--': 0.90, '-2-': 1.10, '--3': 1.30, '12-': 1.45, '1-3': 1.70, '-23': 1.90, '123': 2.30 },
  1: { '---': 0.26, '1--': 0.50, '-2-': 0.65, '--3': 0.90, '12-': 0.85, '1-3': 1.05, '-23': 1.45, '123': 1.55 },
  2: { '---': 0.10, '1--': 0.20, '-2-': 0.30, '--3': 0.35, '12-': 0.40, '1-3': 0.45, '-23': 0.55, '123': 0.75 },
};

function toBaseState(r1: boolean, r2: boolean, r3: boolean): BaseState {
  return `${r1 ? '1' : '-'}${r2 ? '2' : '-'}${r3 ? '3' : '-'}` as BaseState;
}

type Weather = { windTowardOF_mph?: number };
type Batter = { iso?: number; hr?: number; woba?: number; pa?: number; handed?: 'L'|'R' };
type Pitcher = { handed?: 'L'|'R'; pitchCount?: number; timesFacedOrder?: number };

type Park = { hrFactor?: number; runFactor?: number }; // 1.00 = neutral
type Game = { inning: number; half: 'top'|'bot'; awayScore: number; homeScore: number };

export function scoreSituation(params: {
  // state
  outs: 0|1|2;
  r1: boolean; r2: boolean; r3: boolean;
  // actors
  batter: Batter;
  onDeck: Batter; // e.g., Ohtani
  pitcher?: Pitcher;
  // env
  weather?: Weather;
  park?: Park;
  // game
  game: Game;
}) {
  const { outs, r1, r2, r3, batter, onDeck, pitcher, weather, park, game } = params;

  // --- 1) Run Expectancy (RE) ---
  const bs = toBaseState(r1, r2, r3);
  const RE_now = RUN_EXPECTANCY[outs][bs] ?? 0.3; // conservative default
  const RE_norm = clamp(map(RE_now, 0, 2.5, 0, 100), 0, 100);

  // Approx ΔRE if ball is put in play (more impact w/ RISP + <2 outs)
  // Simple heuristic: bases 2&3 with 1 out => large +Δ on single/deep fly
  const rispWeight = (r2 || r3) ? (outs === 0 ? 1.0 : outs === 1 ? 0.8 : 0.25) : 0.25;
  const deltaRE = RE_now * rispWeight; // bounded proxy
  const dRE_norm = clamp(map(deltaRE, 0, 2.0, 0, 100), 0, 100);

  // --- 2) Game Context (inning & closeness) ---
  const inningWeight = game.inning >= 8 ? 1.0 : game.inning >= 7 ? 0.8 : game.inning >= 6 ? 0.6 : 0.4;
  const scoreDiff = Math.abs(game.homeScore - game.awayScore);
  const closeness = scoreDiff === 0 ? 1.0 : scoreDiff === 1 ? 0.85 : scoreDiff === 2 ? 0.6 : 0.35;
  const context = 100 * inningWeight * closeness; // 0–100

  // --- 3) Batter Threat (current PA) ---
  // Normalize ISO + HR rate + wOBA vs league-ish baselines
  const ISO = safe(batter.iso, 0.140);
  const HRr = safe(batter.hr && batter.pa ? batter.hr / batter.pa : undefined, 0.04); // 4% rough
  const wOBA = safe(batter.woba, 0.320);

  const isoScore = clamp(map(ISO, 0.10, 0.30, 0, 100), 0, 100);
  const hrScore  = clamp(map(HRr, 0.02, 0.08, 0, 100), 0, 100);
  const wobaScore= clamp(map(wOBA, 0.280, 0.400, 0, 100), 0, 100);

  // Platoon boost
  const platoonBoost = (batter.handed && pitcher?.handed && batter.handed !== pitcher.handed) ? 1.08 : 1.00;

  const batterThreat = clamp( (0.45*isoScore + 0.35*hrScore + 0.20*wobaScore) * platoonBoost, 0, 100);

  // --- 4) On-Deck Threat (e.g., Ohtani) ---
  // If 1B open & monster on deck → increases *pressure* on current PA (more strikes in zone OR intentional walk dynamics).
  const firstBaseOpen = !r1;
  const onDeckPower = scorePower(onDeck); // 0-100
  let onDeckThreat = onDeckPower * (firstBaseOpen ? 1.00 : 0.50);
  // If base open with 2nd&3rd, the defense may IBB current batter to set up force—BUT if on-deck is elite, that disincentivizes IBB.
  // Model: net effect raises leverage because either outcome spikes run expectancy volatility.
  if (r2 && r3 && firstBaseOpen) onDeckThreat *= 1.15; // spike volatility/leverage

  // --- 5) Weather Boost (tailwind) ---
  const wind = Math.max(0, weather?.windTowardOF_mph ?? 0);
  // Simple linear: +1 mph tailwind → +1.5% HR likelihood (capped 15%)
  const weatherHRBoostPct = Math.min(0.015 * wind, 0.15);
  const weatherScore = clamp(weatherHRBoostPct * 100, 0, 100);

  // --- 6) Park ---
  const pfHR = clamp(map(safe(park?.hrFactor, 1.00), 0.90, 1.10, 0, 100), 0, 100);
  const pfRun= clamp(map(safe(park?.runFactor,1.00), 0.92, 1.08, 0, 100), 0, 100);
  const parkScore = 0.6*pfHR + 0.4*pfRun;

  // --- 7) Pitcher state (optional lightweight) ---
  const fatigue = clamp(map(safe(pitcher?.pitchCount, 12), 50, 95, 0, 100), 0, 100);
  const tto = clamp(map(safe(pitcher?.timesFacedOrder, 1), 2, 3, 0, 100), 0, 100);
  const pitcherState = 0.6*fatiqueOr0(fatigue) + 0.4*tto;

  // --- Composite (weights tuned to be sane) ---
  const composite =
      0.22*RE_norm +
      0.10*dRE_norm +
      0.18*context +
      0.18*batterThreat +
      0.10*onDeckThreat +
      0.10*weatherScore +
      0.07*parkScore +
      0.05*pitcherState;

  // Map composite → Priority (keeps your thresholds)
  const priority = compositeToPriority(composite);

  return {
    priority,
    scores: {
      RE_norm, dRE_norm, context, batterThreat, onDeckThreat, weatherScore, parkScore, pitcherState,
      composite: round(composite,1),
    }
  };
}

// helpers
function safe(v: number|undefined, d: number){ return Number.isFinite(v!) ? (v as number) : d; }
function fatiqueOr0(v: number){ return Number.isFinite(v) ? v : 0; }
function clamp(n: number, a: number, b: number){ return Math.max(a, Math.min(b, n)); }
function map(x: number, inMin: number, inMax: number, outMin: number, outMax: number){
  if (inMax === inMin) return outMin;
  const t = (x - inMin) / (inMax - inMin);
  return outMin + t * (outMax - outMin);
}
function round(n: number, d=0){ const k = Math.pow(10,d); return Math.round(n*k)/k; }

function scorePower(b: Batter){
  const HRr = safe(b.hr && b.pa ? b.hr / b.pa : undefined, 0.04);
  const iso = safe(b.iso, 0.140);
  const s1 = clamp(map(HRr, 0.02, 0.10, 0, 100), 0, 100);
  const s2 = clamp(map(iso, 0.10, 0.35, 0, 100), 0, 100);
  return 0.65*s1 + 0.35*s2;
}

function compositeToPriority(c: number){
  // Smooth mapping to your bands
  if (c >= 90) return 95;   // Critical
  if (c >= 80) return 90;   // High-value
  if (c >= 70) return 85;   // RISP / multi-factor
  if (c >= 60) return 75;   // Notify (Telegram)
  if (c >= 50) return 70;   // Trigger AI analysis
  return 60;                // Standard
}

Keep AI as a refinement, not the source of truth. Send it the numbers you computed and ask for:

event probabilities (Run scores this PA, Extra‑base hit, HR, K, BB)

one-sentence rationale

do not fetch external stats; only reason from provided inputs.

Prompt (server/services/ai-predictions.ts):
