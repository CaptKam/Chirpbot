export type Sport = 'MLB'|'NCAAF';

export type GameTick = {
  sport: Sport;
  gameId: string;
  ts: string;                // ISO
  state: CanonicalState;
  diff: Partial<CanonicalState>;
};

export type CanonicalState = {
  status: 'SCHEDULED'|'LIVE'|'FINAL'|'DELAYED';
  venue: { lat?: number; lon?: number; roof?: 'OPEN'|'CLOSED' };
  score: { home: number; away: number };
  // MLB
  inning?: { half: 'T'|'B'; num: number };
  outs?: 0|1|2;
  bases?: { first: boolean; second: boolean; third: boolean };
  batterId?: string;
  batterStats?: { hrRate?: number; ops?: number };
  // Football
  clock?: { period: number; time?: string };      // period=1..4
  fieldPos?: { yardline?: number; side?: 'HOME'|'AWAY' };
  downDist?: { down?: 1|2|3|4; toGo?: number };
  possession?: 'HOME'|'AWAY';
  situationalFlags?: string[];
  weatherBucket?: string; // e.g., OUT_TO_CF_10_15, INDOOR, CALM
};

export type AlertCandidate = {
  sport: Sport;
  gameId: string;
  type: 'HIGH_SCORING_OPP'|'NINTH_TIE'|'RED_ZONE';
  phase: string;            // "T7" or "Q4_02:15"
  situation: string;        // compact code
  playerId?: string;
  weatherBucket?: string;
  ruleVersion: string;
  score: number;            // priority 0..100
  context: Record<string, unknown>;
};

5) Alert Key & Idempotency (server/models/alert-key.ts)
import { createHash } from 'crypto';
import type { AlertCandidate } from './contracts';

export function buildAlertKey(a: AlertCandidate): string {
  const base = [
    a.sport, a.gameId, a.phase, a.type, a.situation,
    a.playerId ?? '', a.weatherBucket ?? '', a.ruleVersion
  ].join('|');
  return createHash('sha256').update(base).digest('hex').slice(0, 24);
}

7) Engine Flow

server/engines/engine-coordinator.ts (skeleton)

import { normalizeMLB, normalizeNCAAF } from './normalizer';
import { runRules } from './rule-engine';
import { savePlay, tryInsertAlert, isInCooldown, setCooldown } from '../services/db';
import { pushAlert } from '../services/notifier';
import { activeRuleset } from '../services/feature-flags';

export async function processRawTick(sport: 'MLB'|'NCAAF', gameId: string, raw: any) {
  const norm = sport === 'MLB' ? normalizeMLB(raw) : normalizeNCAAF(raw);
  if (norm.state.status !== 'LIVE') return;

  await savePlay(sport, gameId, norm.ts, norm.state.diffSource ?? raw); // append only

  const candidates = runRules(sport, norm.state, activeRuleset());
  for (const c of candidates) {
    const inserted = await tryInsertAlert(c); // ON CONFLICT DO NOTHING
    if (!inserted) continue;                  // dedup hit

    const inCd = await isInCooldown(c);
    if (inCd) continue;                       // suppressed by cooldown

    await pushAlert(c);                       // Telegram/WebSocket
    await setCooldown(c);                     // from config/cooldowns.json
  }
}


server/engines/normalizer.ts (concept)

import type { CanonicalState, GameTick } from '../models/contracts';

export function normalizeMLB(raw: any): GameTick {
  const cs: CanonicalState = {
    status: mapStatus(raw),
    score: { home: raw.homeScore ?? 0, away: raw.awayScore ?? 0 },
    inning: raw.inning ? { half: raw.inning.half, num: raw.inning.num } : undefined,
    outs: raw.outs ?? 0,
    bases: { first: !!raw.on1, second: !!raw.on2, third: !!raw.on3 },
    batterId: raw.batterId,
    batterStats: { hrRate: raw.batterHrRate, ops: raw.batterOps },
    venue: { lat: raw.venue?.lat, lon: raw.venue?.lon, roof: raw.venue?.roof },
    weatherBucket: raw.weatherBucket
  };
  return { sport: 'MLB', gameId: String(raw.gamePk), ts: new Date().toISOString(), state: cs, diff: {} };
}

export function normalizeNCAAF(raw: any): GameTick {
  const cs: CanonicalState = {
    status: mapStatus(raw),
    score: { home: raw.home, away: raw.away },
    clock: { period: raw.period, time: raw.clock }, // "02:15"
    possession: raw.poss,
    fieldPos: { yardline: raw.yardline, side: raw.side },
    downDist: { down: raw.down, toGo: raw.toGo },
    venue: { lat: raw.venue?.lat, lon: raw.venue?.lon, roof: 'OPEN' }
  };
  return { sport: 'NCAAF', gameId: String(raw.id), ts: new Date().toISOString(), state: cs, diff: {} };
}

function mapStatus(r: any): CanonicalState['status'] {
  if (r.status === 'Final') return 'FINAL';
  if (r.status?.includes('Live')) return 'LIVE';
  if (r.status?.includes('Delay')) return 'DELAYED';
  return 'SCHEDULED';
}

8) Two Example Rules

server/engines/rules/mlb.ts – RISP high-scoring opportunity

import type { CanonicalState, AlertCandidate } from '../../models/contracts';
import { buildAlertKey } from '../../models/alert-key';

export function mlbRules_L1(state: CanonicalState, ruleVersion='ruleset_v1'): AlertCandidate[] {
  if (!state.inning) return [];
  const risp = (state.bases?.second || state.bases?.third) === true;
  if (!risp) return [];

  const outs = state.outs ?? 0;
  const phase = `${state.inning.half}${state.inning.num}`; // e.g., T7
  const runners =
    (state.bases?.second ? '2' : '') +
    (state.bases?.third ? '3' : '') || 'NONE';

  const situation = `RISP_${runners}_${outs}OUT`; // e.g., RISP_23_1OUT

  // Simple priority model: more runners + fewer outs -> higher score
  const baseScore = (state.bases?.second?1:0) + (state.bases?.third?1:0);
  let score = 50 + baseScore*20 - outs*10;

  // Weather bump if open roof & out to CF winds bucket recognized
  if (state.weatherBucket?.startsWith('OUT_TO_CF')) score += 10;

  const alert: AlertCandidate = {
    sport: 'MLB',
    gameId: 'unknown', // filled by coordinator
    type: 'HIGH_SCORING_OPP',
    phase,
    situation,
    playerId: state.batterId,
    weatherBucket: state.weatherBucket,
    ruleVersion,
    score,
    context: {
      outs, runners,
      scoreline: state.score
    }
  };
  // the coordinator will compute alert_key and insert
  return [alert];
}


server/engines/rules/ncaaf.ts – Red Zone

import type { CanonicalState, AlertCandidate } from '../../models/contracts';

export function ncaafRules_L1(state: CanonicalState, ruleVersion='ruleset_v1'): AlertCandidate[] {
  if (!state.clock || !state.fieldPos || !state.downDist) return [];
  const yard = state.fieldPos.yardline ?? 99;
  const redZone = yard <= 20;
  if (!redZone) return [];

  const phase = `Q${state.clock.period}_${state.clock.time ?? '??:??'}`;
  const situation = `RZ_${yard}_YDS_D${state.downDist.down ?? 0}_TG${state.downDist.toGo ?? 0}`;
  let score = 60; // base for red zone
  if ((state.clock.period ?? 1) >= 4) score += 10;              // late game bump
  if ((state.downDist.toGo ?? 10) <= 3) score += 10;            // short to-go bump

  const alert: AlertCandidate = {
    sport: 'NCAAF',
    gameId: 'unknown',
    type: 'RED_ZONE',
    phase,
    situation,
    ruleVersion,
    score,
    context: {
      yardline: yard,
      down: state.downDist.down,
      toGo: state.downDist.toGo,
      scoreline: state.score,
      possession: state.possession
    }
  };
  return [alert];
}

3) docs/RUNBOOK.md

# RUNBOOK – Multi-Sport Alerts

## Quick Health
- `GET /healthz` → should be `{ ok: true }`
- `GET /readyz` → `{ db: "ok", redis: "ok", sources: "ok" }`

If `readyz` not OK:
- DB: check `DATABASE_URL`, migrations applied, network allowed.
- Redis: check `REDIS_URL`.
- Sources: API creds/rate limits; try primary + fallback curl.

---

## 1) Users report "a million alerts"
**Symptoms:** Same alert spams Telegram/UI.

**Checklist**
1. **Maintenance switch** (halts delivery, still ingests):
   - set `MAINTENANCE_MODE=1`, redeploy or reload env.
2. **Check dedup**
   - Ensure `alerts` table has `CREATE UNIQUE INDEX ux_alerts_key ON alerts(alert_key);`
   - Confirm `buildAlertKey` uses: `sport|gameId|phase|type|situation|playerId?|weatherBucket?|ruleVersion`
   - Look for `event="alert_deduped"` in logs (should increase).
3. **Cooldowns**
   - Confirm `config/cooldowns.json` loaded and `setCooldown` is called after dispatch.
   - Query cooldowns:
     ```sql
     SELECT * FROM alert_cooldowns WHERE sport='MLB' AND game_id='776553';
     ```
4. **Rule churn**
   - If `alerts_deduped` is high → rules fire on noisy fields.
   - Solution: Only evaluate on relevant `diff` (e.g., bases/outs/yardline changes).
5. **Replay**
   - After fix: `MAINTENANCE_MODE=0`.
   - Optionally re-emit missed alerts via `/admin/replay?gameId=…&since=…`.

---

## 2) No alerts at all
**Checklist**
1. Verify ingest emits ticks (`event="tick"` present).
2. Normalizer maps `status='LIVE'`. If stuck as `SCHEDULED`, your mapping is wrong.
3. Rules enabled: `RULESET_ACTIVE` points to a real ruleset.
4. DB insert errors? Look for `dispatch_fail` or DB connection errors.
5. Cooldown too aggressive? Temporarily set SHORT windows in `config/cooldowns.json`.

---

## 3) Wrong base runners / yardline
**Checklist**
- Log the normalized `CanonicalState` every time the specific fields change.
- Add a **test log** at the fetcher boundary:
  - MLB: `on1,on2,on3,outs,inning`
  - NCAAF: `yardline, down, toGo, period, clock`
- Cross-check the raw source vs normalized values.

---

## 4) Source outage / rate limit
**Checklist**
- Switch to fallback source (already wired in fetcher).
- Reduce poll rate temporarily.
- Keep app running; alerts won’t fire without fresh ticks—this is expected.

---

## 5) Weather looks wrong
**Checklist**
- Roof CLOSED? Force bucket to `INDOOR`.
- Only store **bucket** in Redis; refresh every `WEATHER_REFRESH_MIN` or on wind bucket change.
- If provider jitter causes bucket flapping, introduce hysteresis: require >X mph delta or stable direction for Y minutes.

---

## 6) Rollout / Rollback
- Update logic in new file: `server/engines/rules/*` as `ruleset_vN`.
- Set `RULESET_ACTIVE=ruleset_vN` to roll forward.
- Rollback by restoring previous env value. No code revert needed.

---

## 7) Observability – What to watch
- `alerts_generated`, `alerts_deduped`, `alerts_delivered`, `alerts_suppressed_cooldown`
- Spikes in `source_errors` → data issues upstream.
- High `deduped`/low `delivered` → noisy rules or bad keys.

---

## 8) Local E2E Test
1. Run DB migrations.
2. Start dev: `pnpm dev`
3. In another tab: `pnpm tsx scripts/mock-feed.ts`
4. Expect:
   - MLB: 2–3 alerts max (RISP variants), not repeated back-to-back.
   - NCAAF: 2 alerts (first RZ + later goal-to-go), not repeated.

// scripts/self-check.ts
// Usage: BASE_URL=http://localhost:3000 pnpm tsx scripts/self-check.ts
// Env: DATABASE_URL must be set (same as server)

import { Client } from 'pg';

const BASE_URL = process.env.BASE_URL ?? 'http://localhost:3000';
const DATABASE_URL = process.env.DATABASE_URL;
const WINDOW_MIN = Number(process.env.SELFCHECK_WINDOW_MIN ?? 5);

type Check = { name: string; ok: boolean; info?: string; error?: string };

async function httpJson(path: string) {
  const res = await fetch(`${BASE_URL}${path}`, { method: 'GET' });
  const text = await res.text();
  try { return { status: res.status, body: JSON.parse(text) }; }
  catch { return { status: res.status, body: text }; }
}

async function checkHttp(): Promise<Check[]> {
  const r1 = await httpJson('/healthz');
  const r2 = await httpJson('/readyz');
  const c1: Check = { name: 'healthz', ok: r1.status === 200 && r1.body?.ok === true, info: JSON.stringify(r1.body) };
  const readyOk = r2.status === 200 && typeof r2.body === 'object';
  const sub = readyOk ? Object.entries(r2.body).map(([k,v]) => `${k}:${v}`).join(', ') : String(r2.body);
  const c2: Check = { name: 'readyz', ok: readyOk, info: sub };
  return [c1, c2];
}

async function checkDb(): Promise<Check[]> {
  if (!DATABASE_URL) {
    return [{ name: 'db-connection', ok: false, error: 'DATABASE_URL not set' }];
  }

  const client = new Client({ connectionString: DATABASE_URL });
  await client.connect();

  const q = `
    WITH recent AS (
      SELECT *
      FROM alerts
      WHERE created_at >= now() - INTERVAL '${WINDOW_MIN} minutes'
    )
    SELECT
      count(*)                         AS alerts_total,
      count(*) FILTER (WHERE state='DELIVERED') AS alerts_delivered,
      count(*) FILTER (WHERE state='NEW')       AS alerts_new,
      count(*) FILTER (WHERE state='EXPIRED')   AS alerts_expired,
      count(DISTINCT alert_key)                 AS unique_keys,
      jsonb_agg(jsonb_build_object('type', type, 'n', cnt) ORDER BY cnt DESC) AS by_type
    FROM (
      SELECT type, alert_key, state, created_at, count(*) OVER (PARTITION BY type) AS cnt
      FROM recent
    ) t;
  `;

  const { rows } = await client.query(q);
  await client.end();

  const row = rows[0] ?? {};
  const total = Number(row.alerts_total ?? 0);
  const delivered = Number(row.alerts_delivered ?? 0);
  const unique = Number(row.unique_keys ?? 0);

  const checks: Check[] = [];

  checks.push({
    name: 'db-connection',
    ok: true,
    info: `connected`
  });

  checks.push({
    name: 'alerts-activity',
    ok: total >= 0, // always true, info-only
    info: `last ${WINDOW_MIN}m → total:${total}, delivered:${delivered}, unique_keys:${unique}`
  });

  // Heuristics:
  // - If total >> unique, likely dup churn before DB unique index (shouldn’t happen)
  // - If many NEW and few DELIVERED, notifier may be failing or MAINTENANCE_MODE on
  const dupRatio = unique > 0 ? (total / unique) : 1;
  const info = `dupRatio=${dupRatio.toFixed(2)} (1.0 is ideal), by_type=${JSON.stringify(row.by_type)}`;

  checks.push({
    name: 'duplication',
    ok: dupRatio <= 1.2, // allow slight >1 due to races or retries
    info
  });

  return checks;
}

function printSummary(checks: Check[]) {
  const okAll = checks.every(c => c.ok);
  const lines = checks.map(c =>
    `${c.ok ? '✅' : '❌'} ${c.name}${c.info ? ` — ${c.info}` : ''}${c.error ? ` — ${c.error}` : ''}`
  );
  console.log('\n=== Self-Check Summary ===');
  lines.forEach(l => console.log(l));
  console.log(`Result: ${okAll ? 'PASS' : 'FAIL'}`);

  if (!okAll) {
    console.log('\nTroubleshooting hints:');
    for (const c of checks.filter(x => !x.ok)) {
      if (c.name === 'readyz') {
        console.log('- readyz failed: verify DB/Redis/env and that migrations ran.');
      }
      if (c.name === 'db-connection') {
        console.log('- DB connect failed: check DATABASE_URL and network (does Replit allow external DB?).');
      }
      if (c.name === 'duplication') {
        console.log('- High dupRatio: ensure alerts.alert_key unique index exists and alert_key inputs are stable.');
        console.log('  Also ensure cooldowns are applied after dispatch.');
      }
    }
  }
}

(async () => {
  try {
    const http = await checkHttp();
    const db = await checkDb();
    printSummary([...http, ...db]);
  } catch (err: any) {
    console.error('Self-check crashed:', err?.message ?? err);
    process.exit(1);
  }
})();
