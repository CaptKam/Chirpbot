-- Per-user monitored tennis matches
create table if not exists user_monitored_matches (
  id serial primary key,
  user_id integer references users(id) on delete cascade,
  sport varchar(50) not null default 'TENNIS',
  match_id varchar(64) not null,
  is_monitoring boolean default true,
  created_at timestamp default now(),
  unique(user_id, sport, match_id)
);

-- Tennis alert defaults for new users (toggle in UI later)
-- If you store per-sport JSON in 'settings.alert_types', keep one row per sport.
insert into settings (user_id, sport, alert_types, telegram_enabled)

       jsonb_build_object(
         'breakPoint', true,
         'doubleBreakPoint', true,
         'setPoint', true,
         'matchPoint', true,
         'tiebreakStart', true,
         'momentumSurge', true
       ),
       false
from users u
where not exists (select 1 from settings s where s.user_id=u.id and s.sport='TENNIS');

2) Server: REST endpoints (matches + monitoring + settings)
Routes (Express)
// server/routes/tennis.ts
import { Router } from "express";
import { requireAuth } from "../auth/middleware";
import { tennisAggregator } from "../services/engines/registry"; // see §3
import { pool } from "../db"; // your PG pool

export const tennisRouter = Router();

// Live matches list (for menu selection)
tennisRouter.get("/matches", requireAuth, async (req, res) => {
  const matches = await tennisAggregator.getLiveMatches();
  // annotate with user monitoring flag
  const { rows } = await pool.query(
    "select match_id from user_monitored_matches where user_id=$1 and sport='TENNIS' and is_monitoring=true",
    [req.session.userId]
  );
  const set = new Set(rows.map(r => r.match_id));
  res.json({ matches: matches.map(m => ({ ...m, isMonitoring: set.has(m.matchId) })) });
});

// Toggle monitoring for a match
tennisRouter.post("/toggle_monitoring", requireAuth, async (req, res) => {
  const { matchId, enable } = req.body;
  if (!matchId) return res.status(400).json({ error: "matchId required" });
  await pool.query(
    `insert into user_monitored_matches (user_id, sport, match_id, is_monitoring)
     values ($1,'TENNIS',$2,$3)
     on conflict (user_id, sport, match_id)
     do update set is_monitoring=excluded.is_monitoring`,
    [req.session.userId, matchId, !!enable]
  );
  res.json({ ok: true });
});

}

6) Menu: live matches picker

Create a new page client/src/pages/tennis-matches.tsx:

import { useQuery } from "@tanstack/react-query";

export default function TennisMatchesPage() {
  const { data, isLoading } = useQuery({
    queryKey: ["tennis","matches"],
    queryFn: async () => (await fetch("/api/tennis/matches")).json(),
    refetchInterval: 15_000
  });

  if (isLoading) return <div className="p-4 text-slate-400">Loading matches…</div>;
  const matches = data?.matches ?? [];

  return (
    <div className="p-4 space-y-2">
      {matches.map((m:any) => (
        <div key={m.matchId} className="flex items-center justify-between rounded-lg p-3 bg-white/5">
          <div className="min-w-0">
            <div className="font-semibold text-slate-100 truncate">
              {m.players.home.name} — {m.players.away.name}
            </div>
            <div className="text-xs text-slate-400">
              Set {m.currentSet} • {m.gamesInSet.home}-{m.gamesInSet.away} {m.isTiebreak ? "• TB" : ""}
            </div>
          </div>
          <button
            className={`px-3 py-1 rounded-md text-sm ${m.isMonitoring ? "bg-emerald-500 text-black" : "bg-white/10 text-slate-200"}`}
            onClick={async () => {
              await fetch("/api/tennis/toggle_monitoring", {
                method: "POST",
                headers: {"Content-Type":"application/json"},
                body: JSON.stringify({ matchId: m.matchId, enable: !m.isMonitoring })
              });
            }}
          >
            {m.isMonitoring ? "Monitoring" : "Monitor"}
          </button>
        </div>
      ))}
    </div>
  );
}


Add it to your left menu → “Tennis Matches”.
