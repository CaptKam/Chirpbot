  if (windBoost >= 0.10) {
    const key = `WIND:${g.gameId}:${inningTag(inningNum, half)}`;
    const score = norm100(0.60 + windBoost); // 70..85 typical
    alerts.push({

      message: `Jetstream — tailwind ${(g.weather?.windMph ?? 0)} mph, roof ${g.weather?.roofOpen ? 'open' : 'closed'}.`,

      payload: { windMph: g.weather?.windMph, windOut: g.weather?.windOutToOutfield, roofOpen: g.weather?.roofOpen }

  // ---- 5) HR hitter at bat ----
  if (batterPower >= 1.4) {
    const hrBias = clamp01(0.20 * (batterPower - 1.0) + windBoost); // ~0.08..0.30
    const key = `HRBAT:${g.gameId}:${inningTag(inningNum, half)}:${outs}`;
    const score = norm100(0.55 + hrBias); // 60..85
    alerts.push({
      id: key, alertKey: key, type: 'HR_HITTER_AT_BAT', gameId: g.gameId, sport: 'MLB',
      message: `Power bat up — ${(g.batter?.name ?? 'Batter')}. HR bias ${(hrBias*100).toFixed(0)}%.`,
      score, createdAt: nowIso,
      payload: { batter: g.batter, batterPower, windBoost }
    });
  }


  // ---- 6) Late pressure / close game ----
  const totalRuns = (g.homeScore ?? 0) + (g.awayScore ?? 0);
  const diff = Math.abs((g.homeScore ?? 0) - (g.awayScore ?? 0));
  if (inningNum >= 8 && diff <= 1) {
    const key = `PRESSURE:${g.gameId}:${inningTag(inningNum, half)}`;
    const score = norm100(0.70 + 0.05 * (totalRuns >= 9 ? 1 : 0));
    alerts.push({
      id: key, alertKey: key, type: 'LATE_PRESSURE', gameId: g.gameId, sport: 'MLB',
      message: `Late pressure — ${half} ${inningNum}, one‑run game.`,
      score, createdAt: nowIso,
      payload: { totalRuns, diff }
    });
  }


  // ---- 7) Ninth inning tie ----
  if (inningNum >= 9 && diff === 0) {
    const key = `NINTH_TIE:${g.gameId}:${inningTag(inningNum, half)}`;
    const score = norm100(0.85);
    alerts.push({
      id: key, alertKey: key, type: 'NINTH_TIE', gameId: g.gameId, sport: 'MLB',
      message: `9th‑inning tie — ${g.away} ${g.awayScore ?? 0} • ${g.home} ${g.homeScore ?? 0}.`,
      score, createdAt: nowIso,
      payload: {}
    });
  }


  // ---- 8) Close game late (fallback) ----
  if (inningNum >= 7 && diff <= 1) {
    const key = `CGL:${g.gameId}:${inningTag(inningNum, half)}`;
    const score = norm100(0.65);
    alerts.push({
      id: key, alertKey: key, type: 'CLOSE_GAME_LATE', gameId: g.gameId, sport: 'MLB',
      message: `Close game — late innings within one run.`,
      score, createdAt: nowIso,
      payload: { inningNum, diff }
    });
  }


  return alerts.sort((a, b) => b.score - a.score);
}

Dropped a single-file MLB probability + alerts engine into the canvas (mlb-probability-engine.ts). It’s dead simple and ready to plug into your Replit server:

What it does

Calculates run-scoring probability this half-inning using a compact RE24 table.

Adds modifiers for RISP, wind tailwind / roof, and HR-hitter power.

Emits normalized alerts (0–100) with idempotent keys for deduping.

Includes alert types:

RISP_CHANCE, BASES_LOADED, SCORING_PROBABILITY

WIND_JETSTREAM, HR_HITTER_AT_BAT

LATE_PRESSURE, NINTH_TIE, CLOSE_GAME_LATE

Short answer: right after you build a clean MLBGameState and before you save/emit alerts. Think of it as a stateless scoring pass inside your MLB engine tick.

Here’s the flow with the exact insertion point:

Fetch & merge feeds → (StatsAPI/ESPN/etc.)

Normalize → extractGameState() → MLBGameState

➡️ INSERT HERE → call evaluateGameForAlerts(gameState)

Dedupe & upsert alerts (unique on alertKey)

Publish → /admin socket (alert.new) + optional Telegram/webhooks

Persist game snapshot / health

server/services/engines/mlb-engine.ts

After your extractGameState() and only if status === 'LIVE'.

// mlb-engine.ts (tick for a single game)
import { evaluateGameForAlerts } from "../mlb-probability-engine";
import { prisma } from "../db";
import { ioAdmin } from "../sockets/admin"; // io.of('/admin')

export async function processLiveMLBGame(rawFeeds: any) {
  const game = extractGameState(rawFeeds);            // 1) normalized MLBGameState
  await upsertGameState(game);                        // optional snapshot save

  if (game.status !== "LIVE") return;

  const alerts = evaluateGameForAlerts(game);         // 2) ▶️ INSERTED HERE

  // 3) dedupe + upsert (idempotent on alertKey)
  for (const a of alerts) {
    try {
      await prisma.alert.upsert({
        where: { alertKey: a.alertKey },
        update: { status: "OPEN", payload: a.payload, score: a.score, updatedAt: new Date(a.createdAt) },
        create: {
          alertKey: a.alertKey,
          type: a.type,
          sport: "MLB",
          gameId: a.gameId,
          message: a.message,
          score: a.score,
          payload: a.payload,
          createdAt: new Date(a.createdAt)
        }
      });

      // 4) emit to admin UI (non-blocking)
      ioAdmin.emit("alert.new", {
        id: a.alertKey,
        sport: "MLB",
        type: a.type,
        gameId: a.gameId,
        message: a.message,
        score: a.score,
        payload: a.payload,
        createdAt: a.createdAt,
      });

      // 5) (optional) notify channels gated by user permissions
      // await fanOutToTelegramIfAllowed(a);
    } catch (e) {
      // swallow unique violations safely; log others
      console.error("alert upsert err", a.alertKey, e);
    }
  }

  // 6) always emit latest game snapshot for your Live tab
  ioAdmin.emit("game.update", {
    sport: "MLB",
    extGameId: game.gameId,
    status: game.status,
    home: game.home, away: game.away,
    homeScore: game.homeScore, awayScore: game.awayScore,
    inning: game.inning, outs: game.outs, runners: game.runners,
    updatedAt: game.timestampIso || new Date().toISOString()
  });
}

Same insertion: immediately after you have a normalized MLBGameState.

Pros: isolates side effects; Cons: one more component.

Inputs the engine needs ready before the call

runners (1B/2B/3B), outs, inning {number, half}, scores

batter basics (HR, PA, ISO if you have it)

Weather enrichment (roofOpen, windOutToOutfield, windMph)

Do that enrichment in extractGameState() so evaluateGameForAlerts() can stay pure & cheap.

Cooldown / spam control (optional)

Even with idempotent keys, you may want a soft cooldown per alert type:

Keep a short-lived cache {alertKey -> lastSentAt} (e.g., 30–60s) to skip noisy repeats when feeds jitter.

Tie-in to your admin permissions

When you go to fan-out alerts (Telegram/webhooks):

Check isAlertAllowed(userId, alertTypeKey) from your UserAlertPermission table before sending.

The admin UI still shows all generated alerts; delivery is what’s filtered.

TL;DR: Call the probability engine inside your MLB engine tick, immediately after state normalization. Upsert alerts with a unique alertKey, then emit to the /admin socket (and any channels permitted). That’s the safest, simplest insertion point and it plugs straight into the MLB Live tab you already have.
