# Admin Portal Scaffold (RBAC + AI Controls + Per‑Sport Rules)

Below is a minimal, production‑oriented scaffold you can paste into your app. It adds:

* **RBAC** (Admin/Manager/Analyst/Viewer)
* **AI Controls** per team & sport (backend + UI)
* **Rules Engine** entities (RuleSet/Rule) with JSONLogic style
* **API routes** (Express)
* **React admin pages** (shadcn/ui + Tailwind) for AI Controls & Rules

> All examples assume: Express backend (`server/…`), Prisma for DB, React client.

---

## 2) RBAC Middleware (`server/middleware/rbac.ts`)

```ts
import { Request, Response, NextFunction } from "express";

type Role = "ADMIN" | "MANAGER" | "ANALYST" | "VIEWER";

export function requireRole(minRole: Role) {
  const rank: Record<Role, number> = { ADMIN: 3, MANAGER: 2, ANALYST: 1, VIEWER: 0 };
  return (req: Request, res: Response, next: NextFunction) => {
    const user = (req as any).user; // set by your auth middleware
    if (!user) return res.status(401).json({ error: "unauthenticated" });
    const ok = rank[user.role as Role] >= rank[minRole];
    if (!ok) return res.status(403).json({ error: "forbidden" });
    next();
  };
}
```

## 3) AI Controller (single funnel) (`server/services/ai-controller.ts`)

```ts
import { analyzeAlert } from "../services/openai";

export type AiSettings = {
  enabled: boolean;
  dryRun: boolean;
  rateLimitMs: number;
  minProbability: number; // 0..1
  inningThreshold: number;
  allowTypes: string[];
  redactPii: boolean;
  model?: string;
  maxTokens?: number;
};

let lastCallAt = 0;

function redact(input: any, on: boolean){
  if (!on) return input;
  const c = JSON.parse(JSON.stringify(input));
  delete c.playerId; delete c.pitcherId; delete c.batterId; // example
  if (c.batterName) c.batterName = "REDACTED";
  if (c.pitcherName) c.pitcherName = "REDACTED";
  return c;
}

export async function maybeEnhanceAlert(alert: any, settings: AiSettings){
  if (!settings.enabled) return alert;

  const now = Date.now();
  if (settings.rateLimitMs && now - lastCallAt < settings.rateLimitMs) return alert;
  if (alert.probability < settings.minProbability) return alert;
  if (typeof alert.inning === "number" && alert.inning < settings.inningThreshold) return alert;
  if (settings.allowTypes?.length && !settings.allowTypes.includes(alert.type)) return alert;

  lastCallAt = now;
  const context = redact(alert.context ?? {}, settings.redactPii);
  const ai = await analyzeAlert({
    sport: alert.sport, type: alert.type, probability: alert.probability,
    inning: alert.inning, inningState: alert.inningState, context
  }, { maxTokens: settings.maxTokens });

  if (settings.dryRun) { console.log("[AI dry-run]", ai); return alert; }
  return { ...alert, ai };
}
```

> Plug this in **after** your dedup check and **before** broadcast.

## 4) Admin API Routes (Express)

**`server/routes/admin-ai.ts`**

```ts
import { Router } from "express";
import { requireRole } from "../middleware/rbac";
import { prisma } from "../prisma"; // your Prisma client

export const adminAi = Router();

// Read per-sport AI settings
adminAi.get("/teams/:teamId/sports/:sportId/ai-settings", requireRole("MANAGER"), async (req, res) => {
  const { teamId, sportId } = req.params;
  const s = await prisma.aISettings.findUnique({ where: { teamId_sportId: { teamId, sportId } } });
  res.json(s);
});

// Update per-sport AI settings
adminAi.patch("/teams/:teamId/sports/:sportId/ai-settings", requireRole("MANAGER"), async (req, res) => {
  const { teamId, sportId } = req.params;
  const body = req.body;
  // TODO: validate body ranges
  const s = await prisma.aISettings.upsert({
    where: { teamId_sportId: { teamId, sportId } },
    create: { teamId, sportId, ...body },
    update: { ...body }
  });
  res.json(s);
});
```

**`server/routes/admin-rules.ts`**

```ts
import { Router } from "express";
import { requireRole } from "../middleware/rbac";
import { prisma } from "../prisma";

export const adminRules = Router();

// List rulesets for a sport
adminRules.get("/teams/:teamId/sports/:sportId/rulesets", requireRole("ANALYST"), async (req, res) => {
  const { teamId, sportId } = req.params;
  const sets = await prisma.ruleSet.findMany({ where: { teamId, sportId }, include: { rules: true } });
  res.json(sets);
});

// Create draft ruleset
adminRules.post("/teams/:teamId/sports/:sportId/rulesets", requireRole("ANALYST"), async (req, res) => {
  const { teamId, sportId } = req.params;
  const { name } = req.body;
  const set = await prisma.ruleSet.create({ data: { teamId, sportId, name, status: "DRAFT" } });
  res.json(set);
});

// Update ruleset metadata (rename/status)
adminRules.patch("/rulesets/:id", requireRole("MANAGER"), async (req, res) => {
  const { id } = req.params; const body = req.body;
  const set = await prisma.ruleSet.update({ where: { id }, data: body });
  res.json(set);
});

// Publish ruleset
adminRules.post("/rulesets/:id/publish", requireRole("MANAGER"), async (req, res) => {
  const { id } = req.params;
  const current = await prisma.ruleSet.findUnique({ where: { id } });
  if (!current) return res.status(404).json({ error: "not found" });
  const nextVersion = (current.version ?? 1) + 1;
  const published = await prisma.ruleSet.update({ where: { id }, data: { status: "ACTIVE", version: nextVersion, publishedAt: new Date() } });
  res.json(published);
});

// CRUD rules in a ruleset
adminRules.post("/rulesets/:rulesetId/rules", requireRole("ANALYST"), async (req, res) => {
  const { rulesetId } = req.params; const data = req.body;
  const rule = await prisma.rule.create({ data: { rulesetId, ...data } });
  res.json(rule);
});

adminRules.patch("/rules/:id", requireRole("ANALYST"), async (req, res) => {
  const { id } = req.params; const data = req.body;
  const rule = await prisma.rule.update({ where: { id }, data });
  res.json(rule);
});

adminRules.delete("/rules/:id", requireRole("MANAGER"), async (req, res) => {
  const { id } = req.params;
  await prisma.rule.delete({ where: { id } });
  res.json({ ok: true });
});
```

## 5) Rules Evaluation (JSONLogic + safe function map)

**`server/services/rules-engine.ts`**

```ts
import jsonLogic from "json-logic-js";
import { computeProb, priorityFrom } from "../services/math"; // your helpers

const fn = {
  computeProb: (frame:any, ctx:any) => computeProb(frame.runners, frame.outs, ctx),
  priorityFrom: (prob:number, inning:number, scoreDiff:number) => priorityFrom(prob, inning, scoreDiff),
};

export function evaluateRule(rule:any, frame:any, ctx:any){
  const ok = jsonLogic.apply(rule.conditionJson, frame);
  if (!ok) return null;

  function resolve(node:any){
    if (node && typeof node === "object" && node.fn && Array.isArray(node.args)){
      const f = (fn as any)[node.fn]; if (!f) throw new Error(`Forbidden fn: ${node.fn}`);
      const args = node.args.map((a:any)=> (a === "probability" ? out.probability : (typeof a === "string" && a.startsWith("ctx.")) ? a.split('.').slice(1).reduce((acc:any,k)=>acc?.[k], ctx) : a));
      return f(...args);
    }
    return node;
  }

  const out:any = JSON.parse(JSON.stringify(rule.actionJson));
  for (const k of Object.keys(out)) out[k] = resolve(out[k]);
  return out;
}
```

```tsx
import { useEffect, useState } from "react";
import { Button } from "@/components/ui/button";
import { Switch } from "@/components/ui/switch";
import { api } from "@/lib/api";

export function RulesEditor({ teamId, sportId }: { teamId:string; sportId:string }){
  const [sets, setSets] = useState<any[]>([]);
  const [selected, setSelected] = useState<any | null>(null);

  useEffect(()=>{ load(); },[teamId,sportId]);
  async function load(){ const r = await api.get(`/api/admin/teams/${teamId}/sports/${sportId}/rulesets`); setSets(r||[]); setSelected(r?.find((x:any)=> x.status==='ACTIVE') || r?.[0] || null); }

  async function createSet(){ const name = prompt("Ruleset name?") || `Rules ${Date.now()}`; await api.post(`/api/admin/teams/${teamId}/sports/${sportId}/rulesets`, { name }); await load(); }

      <div className="space-y-3">

          <Button variant="secondary" onClick={createSet}>New</Button>
        </div>
        <div className="space-y-2 max-h-[60vh] overflow-auto">
          {sets.map((s)=> (
            <div key={s.id} onClick={()=> setSelected(s)} className={`p-3 rounded-md border cursor-pointer ${selected?.id===s.id? 'bg-muted' : ''}`}>
              <div className="font-medium">{s.name}</div>
              <div className="text-xs text-muted-foreground">{s.status} • v{s.version}</div>
            </div>
          ))}
        </div>
      </div>

      <div className="md:col-span-2 space-y-4">
        {!selected ? <div className="text-sm text-muted-foreground">Select a ruleset</div> : (
          <RulesList ruleset={selected} onChanged={load} />
        )}
        <div className="flex gap-3">
          <Button onClick={publish} disabled={!selected}>Publish Active</Button>
        </div>
      </div>
    </div>
  );
}

function RulesList({ ruleset, onChanged }:{ ruleset:any; onChanged:()=>void }){
  const [local, setLocal] = useState(ruleset.rules || []);
  useEffect(()=> setLocal(ruleset.rules||[]), [ruleset]);

  async function toggle(rule:any){
    await api.patch(`/api/admin/rules/${rule.id}`, { enabled: !rule.enabled });
    onChanged();
  }

  return (
    <div className="space-y-3">
      {(local||[]).sort((a:any,b:any)=> a.order-b.order).map((r:any)=> (
        <div key={r.id} className="p-3 rounded-md border">
          <div className="flex items-center justify-between">
            <div>
              <div className="font-medium">{r.name} <span className="text-xs text-muted-foreground">(L{r.level})</span></div>
              <div className="text-xs text-muted-foreground">prio {r.priority} • order {r.order}</div>
            </div>
            <Switch checked={r.enabled} onCheckedChange={()=> toggle(r)} />
          </div>
          <pre className="mt-3 text-xs bg-muted p-2 rounded-md overflow-auto max-h-40">{JSON.stringify(r.conditionJson, null, 2)}</pre>
          <pre className="mt-2 text-xs bg-muted p-2 rounded-md overflow-auto max-h-40">{JSON.stringify(r.actionJson, null, 2)}</pre>
        </div>
      ))}
    </div>
  );
}
```

## 7) Wiring in the Engine

* Load **active RuleSet** for sport/team at engine start (cache 30s).
* Evaluate rules → build alert payloads → **dedup** → **maybeEnhanceAlert** (AI) → persist/broadcast.

```ts
import { evaluateRule } from "../services/rules-engine";
import { maybeEnhanceAlert } from "../services/ai-controller";

async function handleFrame(teamId:string, sportId:string, frame:any, ctx:any){
  const ruleset = await getActiveRuleset(teamId, sportId); // cache
  for (const rule of ruleset.rules.filter((r:any)=> r.enabled)){
    const out = evaluateRule(rule, frame, ctx);
    if (!out) continue;
    const alert = {
      sport: sportId,
      type: out.type,
      probability: out.probability,
      priority: out.priority,
      inning: frame.inning,
      inningState: frame.half,
      context: { frameSummary: summarize(frame) }
    };
    if (!dedup.shouldEmit(`${out.type}`, toDedupFrame(frame))) continue;
    const settings = await getAiSettings(teamId, sportId);
    const finalAlert = await maybeEnhanceAlert(alert, settings);
    await persistAndBroadcast(finalAlert);
  }
}
```
