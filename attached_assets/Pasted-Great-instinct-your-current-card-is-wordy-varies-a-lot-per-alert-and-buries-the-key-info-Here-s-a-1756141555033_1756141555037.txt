type Runners = { first?: boolean; second?: boolean; third?: boolean };

function inningLabel(half?: string, inning?: number) {
  if (!inning) return "";
  const h = (half || "").toLowerCase();
  return `${h.startsWith("top") ? "Top" : h.startsWith("bottom") ? "Bot" : ""} ${inning}`;
}

function outsLabel(outs?: number) {
  if (outs == null) return "";
  return `${outs} ${outs === 1 ? "out" : "outs"}`;
}

function pct(n?: number) {
  if (n == null) return "";
  const val = n > 1 ? n : n * 100;
  return `${Math.round(val)}%`;
}

function Pill({ children, className }: { children: React.ReactNode; className?: string }) {
  return (
    <span className={cn(
      "inline-flex items-center gap-1 rounded-full px-2 py-0.5 text-[11px] font-semibold",
      "bg-white/7 ring-1 ring-white/15 text-slate-200",
      className
    )}>
      {children}
    </span>
  );
}

function RunnersDiamond({ r }: { r: Runners }) {
  // tiny diamond made of 4 squares; highlight 1B/2B/3B as filled
  return (
    <div className="relative w-10 h-10">
      {/* 2B */}
      <div className={cn(
        "absolute left-1/2 top-0 -translate-x-1/2 w-3 h-3 rotate-45 border border-white/30",
        r.second && "bg-yellow-300/90 border-yellow-300/90"
      )}/>
      {/* 3B */}
      <div className={cn(
        "absolute left-0 top-1/2 -translate-y-1/2 w-3 h-3 rotate-45 border border-white/30",
        r.third && "bg-yellow-300/90 border-yellow-300/90"
      )}/>
      {/* 1B */}
      <div className={cn(
        "absolute right-0 top-1/2 -translate-y-1/2 w-3 h-3 rotate-45 border border-white/30",
        r.first && "bg-yellow-300/90 border-yellow-300/90"
      )}/>
      {/* Home */}
      <div className="absolute left-1/2 bottom-0 -translate-x-1/2 w-3 h-3 rotate-45 border border-white/20"/>
    </div>
  );
}

function situationLabel(a: Alert) {
  const g = a.gameInfo;
  const risp = g?.runners?.second || g?.runners?.third;
  const bases =
    (g?.runners?.second && g?.runners?.third) ? "2nd & 3rd" :
    g?.runners?.third ? "3rd" :
    g?.runners?.second ? "2nd" :
    g?.runners?.first ? "On Base" : "Bases Empty";
  if ((a.type || "").toLowerCase().includes("power_hitter")) {
    return `Power Hitter · ${bases} · ${outsLabel(g?.outs)}`;
  }
  if ((a.type || "").toLowerCase().includes("risp")) {
    return `RISP · ${bases} · ${outsLabel(g?.outs)}`;
  }
  if ((a.type || "").toLowerCase().includes("bases loaded")) {
    return `Bases Loaded · ${outsLabel(g?.outs)}`;
  }
  return `${(a.type || "").replace(/_/g, " ") } · ${outsLabel(g?.outs)}`;
}

1) Define a sport-agnostic view model

Put this near your types (e.g., client/src/lib/alert-vm.ts):

export type Sport = 'MLB'|'NFL'|'NBA'|'NHL'|'SOCCER'|'TENNIS';

export type AlertVM = {
  id: string;
  sport: Sport;
  title: string;                 // e.g., "RISP", "Red Zone", "Power Play"
  situation: string;             // e.g., "2nd & 3rd • 1 out", "1st & 10 @ OPP 22"
  scoreline: string;             // "LAD 3 — NYM 2"
  period: string;                 // "Top 7", "Q4 4:12", "3rd 08:22", "Set 2 • 30-40"
  edge: { label: string; value: string }; // RP/WP/xG/TD% etc. e.g., {label:"RP", value:"71%"}
  priority: number;
  actionLine?: string;           // "LIVE BET: Over 0.5 runs this inning"
  tags?: string[];               // ["Wind +18%","Tier A","Red Zone"]
  isNew?: boolean;
  createdAt?: string;
  // Optional sport-specific mini viz (runners diamond, red-zone bar, possession chip...)
  widget?: React.ReactNode;
  actor?: string;                // e.g., "Ohtani", "Mahomes"
};

2) Create a tiny adapter layer per sport

Adapters convert your raw alert record → AlertVM. Keep them in client/src/adapters/.

// adapters/base.ts
import type { AlertVM, Sport } from "@/lib/alert-vm";
export interface SportAdapter {
  sport: Sport;
  canHandle: (a: any) => boolean;          // a.sport === 'MLB', etc.
  toViewModel: (a: any) => AlertVM;
}
export const adapters: SportAdapter[] = [];
export function toVM(alert: any): AlertVM {
  const ad = adapters.find(x => x.canHandle(alert));
  if (!ad) throw new Error(`No adapter for sport ${alert.sport}`);
  return ad.toViewModel(alert);
}

MLB example (uses your existing data)
// adapters/mlb.tsx
import { adapters, SportAdapter } from "./base";
import type { AlertVM } from "@/lib/alert-vm";
import { RunnersDiamond } from "@/components/RunnersDiamond"; // from your new UI

const mlb: SportAdapter = {
  sport: "MLB",
  canHandle: a => a.sport === "MLB",
  toViewModel: (a: any): AlertVM => {
    const g = a.gameInfo || {};
    const bases =
      g?.runners?.second && g?.runners?.third ? "2nd & 3rd" :
      g?.runners?.third ? "3rd" :
      g?.runners?.second ? "2nd" :
      g?.runners?.first ? "On Base" : "Bases Empty";

    const period = `${(g.inningState||'').toLowerCase().startsWith('top')?'Top':'Bot'} ${g.inning ?? ''}`.trim();
    const edgeVal = a.gameInfo?.scoringProbability != null
      ? `${Math.round(((a.gameInfo.scoringProbability > 1 ? a.gameInfo.scoringProbability/100 : a.gameInfo.scoringProbability) * 100))}%`
      : a.probability != null ? `${Math.round((a.probability>1?a.probability/100:a.probability)*100)}%` : '';

    return {
      id: a.id,
      sport: "MLB",
      title: (a.type || '').replace(/_/g,' '),
      situation: `${bases} • ${g.outs ?? 0} ${g.outs===1?'out':'outs'}`,
      scoreline: `${g.awayTeam} ${g.score?.away} — ${g.homeTeam} ${g.score?.home}`,
      period,
      edge: { label: (a.type||'').toLowerCase().includes('risp') ? "RP" : "Edge", value: edgeVal },
      priority: a.priority ?? 70,
      actionLine: a.description || "LIVE BET: High-leverage moment.",
      tags: [
        g.tier ? `Tier ${g.tier}` : null,
        g.windBoost ? `Wind +${Math.round((g.windBoost-1)*100)}%` : null
      ].filter(Boolean) as string[],
      isNew: !a.seen,
      createdAt: a.createdAt || a.created_at,
      widget: <RunnersDiamond r={g.runners ?? {}} />
    };
  }
};

NFL example (how it maps into the same card)
// adapters/nfl.tsx
import { adapters, SportAdapter } from "./base";
import type { AlertVM } from "@/lib/alert-vm";
import { RedZoneBar } from "@/components/RedZoneBar"; // simple 0–20 yds progress bar

const nfl: SportAdapter = {
  sport: "NFL",
  canHandle: a => a.sport === "NFL",
  toViewModel: (a: any): AlertVM => {
    const g = a.gameInfo || {};
    const spot = g.yardLine != null ? `@ ${g.side ?? 'OPP'} ${g.yardLine}` : '';
    const downDist = g.down && g.distance ? `${g.down}&${g.distance}` : '';
    const period = `Q${g.quarter ?? '?'} ${g.clock ?? ''}`.trim();
    // Use TD%/Score%/WP depending on the alert type; here we pick TD%
    const edgeVal = g.tdProbability != null ? `${Math.round(g.tdProbability*100)}%` : '';
    return {
      id: a.id,
      sport: "NFL",
      title: (a.type || '').replace(/_/g,' '),       // e.g., "Red Zone", "Two-Minute Drill"
      situation: [downDist, spot].filter(Boolean).join(' • '),
      scoreline: `${g.awayTeam} ${g.score?.away} — ${g.homeTeam} ${g.score?.home}`,
      period,
      edge: { label: "TD%", value: edgeVal },
      priority: a.priority ?? 80,
      actionLine: a.description || "LIVE BET: Drive in scoring range.",
      tags: [g.redZone ? "Red Zone" : null, g.noTimeouts ? "No TOs" : null].filter(Boolean) as string[],
      isNew: !a.seen,
      createdAt: a.createdAt || a.created_at,
      widget: <RedZoneBar yardsToGoal={g.yardsToGoal ?? 20} />
    };
  }
};
adapters.push(nfl);
