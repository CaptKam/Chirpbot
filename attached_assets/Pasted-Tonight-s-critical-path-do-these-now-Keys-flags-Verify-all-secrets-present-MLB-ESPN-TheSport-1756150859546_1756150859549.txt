Code-level guardrails (small but decisive)

Polling concurrency gate

let inFlight = false;
async function tick() {
  if (inFlight) return;
  inFlight = true;
  try { await aggregateAndProcess(); }
  finally { inFlight = false; }
}


Live status mapping

const LIVE = (s: { abstractGameState?: string; codedGameState?: string }) =>
  s.abstractGameState === 'Live' || s.codedGameState === 'I';


Wind carry (stadium-aware)

function carry(cfAzimuthDeg:number, windDirDeg:number, mph:number) {
  let d = Math.abs(windDirDeg - cfAzimuthDeg) % 360;
  if (d > 180) d = 360 - d;
  const outMph = Math.max(0, mph * Math.cos(d * Math.PI / 180));
  return 1 + 0.018 * outMph;
}
