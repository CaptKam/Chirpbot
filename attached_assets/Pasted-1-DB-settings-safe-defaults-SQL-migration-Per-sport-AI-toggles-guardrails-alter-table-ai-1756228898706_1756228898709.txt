3) Heuristic gate (when to ask AI)

We only call AI when there’s real leverage before the obvious alert fires.

server/services/engines/tennis-opportunity-gate.ts

import type { TennisMatch } from "../sources/tennis-sources";

export function stablePointKey(s: TennisMatch) {
  const set = s.currentSet ?? 0;
  const gH = s.gamesInSet?.home ?? 0, gA = s.gamesInSet?.away ?? 0;
  const pH = s.points?.home ?? "-", pA = s.points?.away ?? "-";
  const tbH = s.tbPoints?.home ?? 0, tbA = s.tbPoints?.away ?? 0;
  const sv = s.server ?? "-";
  const tb = s.isTiebreak ? `TB:${tbH}-${tbA}` : `P:${pH}-${pA}`;
  return `S${set}:G${gH}-${gA}:${tb}:SV${sv}`;
}

export function leverageScore(s: TennisMatch) {
  const gMax = Math.max(s.gamesInSet.home, s.gamesInSet.away);
  let L = 0;
  if (gMax >= 9) L += 2;                        // late in set
  if ((s.setsWon.home === 1 && s.setsWon.away === 1)) L += 2; // deciding set (Bo3)
  if (s.isTiebreak) L += 3;
  return L; // 0..7
}

/** True only in pre-alert tension states (let AI “look ahead”). */
export function shouldProbeAI(s: TennisMatch) {
  if (!s.points && !s.isTiebreak) return false; // need some point context
  const L = leverageScore(s);
  // Pre-BP pressure patterns
  const p = s.points;
  const sv = s.server;
  const preBP =
    (!s.isTiebreak && p && sv && (
      // receiver-friendly scores: 0-30, 15-30, 30-30 on server's game
      (p[sv] === "0" && (p[opp(sv)] === "30")) ||
      (p[sv] === "15" && p[opp(sv)] === "30") ||
      (p[sv] === "30" && p[opp(sv)] === "30")
    ));
  // Late tiebreak swing moments
  const lateTB = s.isTiebreak && ((s.tbPoints?.home ?? 0) + (s.tbPoints?.away ?? 0)) >= 8;

  return (L >= 2 && (preBP || lateTB));
}

function opp(side: "home"|"away") { return side === "home" ? "away" : "home"; }

4) OpenAI call (short, novice-friendly output)

server/services/ai/tennis-opportunity.ts

import OpenAI from "openai";
import type { TennisMatch } from "../sources/tennis-sources";

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY! });

export type TennisOppResult = { text: string; priority: number; confidence: number };

export async function makeTennisOpportunityCopy(
  s: TennisMatch,
  opts: {
    noviceMode: boolean;
    minPriority: number;          // opp_min_priority
    model: string;                // e.g., "gpt-4o-mini"
    maxTokens?: number;
  }
): Promise<TennisOppResult> {
  const players = `${s.players.home.name} vs ${s.players.away.name}`;
  const setStr = `Set ${s.currentSet}`;
  const gameStr = `${s.gamesInSet.home}-${s.gamesInSet.away}`;
  const ptStr = s.isTiebreak
    ? `TB ${s.tbPoints?.home ?? 0}-${s.tbPoints?.away ?? 0}`
    : `${s.points?.home ?? "—"}-${s.points?.away ?? "—"}`;
  const pov = s.server ? (s.server === "home" ? s.players.away.name : s.players.home.name) : "";

  const sys = [
    "You are a cautious tennis trading assistant.",
    "Spot SHORT-TERM opportunities (next 1–3 points) and explain for a novice.",
    "Never guarantee outcomes. Avoid brand odds. Keep it educational.",
    "Output STRICTLY 2 lines:",
    "1) AI WATCH: <opportunity in 8–12 words>",
    "2) Why: <one stat/fact in 10–14 words>",
  ].join(" ");

  const user = [
    `Match: ${players}`,
    `${setStr} • Games ${gameStr} • ${s.isTiebreak ? "Tiebreak" : "Points"} ${ptStr}`,
    s.server ? `Server: ${s.server === "home" ? s.players.home.name : s.players.away.name}` : "Server: —",
    `Novice mode: ${opts.noviceMode ? "on" : "off"}`
  ].join("\n");

  const resp = await openai.chat.completions.create({
    model: opts.model,
    temperature: 0.3,
    max_tokens: opts.maxTokens ?? 90,
    messages: [
      { role: "system", content: sys },
      { role: "user", content: user }
    ]
  });

  const raw = resp.choices[0]?.message?.content?.trim() || "";
  // Basic confidence heuristic from wording; you can refine with a logit_bias pattern or post-score
  const confidence = raw.toLowerCase().includes("likely") ? 0.72 :
                     raw.toLowerCase().includes("very good") ? 0.8 : 0.65;

  // Map to priority band
  const priority = Math.max(opts.minPriority, Math.round( (confidence * 100 - 20) ));
  return { text: raw, priority, confidence };
}


Output is always two short lines—perfect for your 3-second card read and novice mode.

5) TennisEngine integration (non-blocking, cooldowned)

server/services/engines/tennis-engine.ts – add the AI lane at the end of each match tick, before a point becomes BP/SP/MP.

import { shouldProbeAI, stablePointKey } from "./tennis-opportunity-gate";
import { makeTennisOpportunityCopy } from "../ai/tennis-opportunity";

const aiCooldown = new Map<string, number>(); // key = `${matchId}:${set}`

async function maybeEmitAIOpportunity(s: TennisMatch, settings: any) {
  if (!settings?.betting_opps_enabled) return;
  if (!settings?.novice_mode) settings.novice_mode = true;

  // Per-match cooldown
  const k = `${s.matchId}:${s.currentSet}`;
  const now = Date.now();
  const last = aiCooldown.get(k) ?? 0;
  if (now - last < (settings.opp_cooldown_ms ?? 90000)) return;

  if (!shouldProbeAI(s)) return;

  // Compose dedup key on stable point id so escalations update in place
  const dedupKey = `${s.matchId}:TENNIS_AI_OPPORTUNITY:${stablePointKey(s)}`;

  if (!alertDeduplication.shouldAllow("TENNIS_AI_OPPORTUNITY", s.matchId, dedupKey, {
    escalationOnly: true, timeWindow: 20000, realertAfterMs: settings.opp_cooldown_ms ?? 90000
  })) return;

  // OpenAI (non-blocking for the rest of the engine)
  let text = "";
  let priority = settings.opp_min_priority ?? 88;
  let confidence = 0.65;

  try {
    const r = await makeTennisOpportunityCopy(s, {
      noviceMode: !!settings.novice_mode,
      minPriority: priority,
      model: settings.model ?? "gpt-4o-mini",
      maxTokens: 90
    });
    text = r.text;
    priority = r.priority;
    confidence = r.confidence;
  } catch (e) {
    // Fallback copy if AI fails
    const tb = s.isTiebreak ? `TB ${s.tbPoints?.home ?? 0}-${s.tbPoints?.away ?? 0}` : "";
    text = `AI WATCH: Pressure point likely next\nWhy: Score favors receiver; leverage high ${tb ? "in tiebreak" : "late in set"}`;
    priority = Math.max(priority, 88);
  }

  // Respect priority floor
  if (priority < (settings.opp_min_priority ?? 88)) return;

  const alert = {
    id: crypto.randomUUID(),
    type: "TENNIS_AI_OPPORTUNITY",
    title: "AI WATCH",
    description: text,
    priority,
    sport: "TENNIS",
    game_info: {
      matchId: s.matchId,
      set: s.currentSet,
      games: s.gamesInSet,
      points: s.isTiebreak ? s.tbPoints : s.points,
      isTiebreak: s.isTiebreak,
      server: s.server
    },
    created_at: new Date().toISOString()
  };

  await storage.createAlert(alert);
  this.onAlert?.({ type: "alert", data: alert });   // WS broadcast
  try { await sendTelegramAlert(this.telegramCfg, alert); } catch {}

  aiCooldown.set(k, now);
}


Call await maybeEmitAIOpportunity(s, aiSettingsForTennis) in your TennisEngine tick after you update momentum and before or after standard checks (it’s independent).

6) Admin & user controls

Admin panel (Tennis AI): switches & inputs bound to ai_settings:

betting_opps_enabled (master on/off)

opp_min_priority (default 88)

opp_cooldown_ms (90–180s good)

opp_max_per_set (soft cap; optional)

novice_mode (true → simpler wording)

model, daily_token_cap

User Settings → Tennis: AI Opportunities toggle (maps to settings.alert_types.aiOpportunity).

If off, you can still log to DB but don’t fan-out to that user.

7) UI: make it obvious and novice-friendly

Card: pill “AI WATCH”, two lines from the model, small “?” tooltip (“Not betting advice. Educational only.”).

Toast policy: only toast if priority ≥ opp_min_priority (e.g., 90); otherwise feed-only.

Batching: per-match 5s window like other alerts; escalation updates the existing card (dedup key uses stablePointKey).

Safety & behavior notes

This lane is decorative: it must never block or delay normal alerts.

Keep copies educational (“watch for”, “pressure point”, “in next 1–3 points”).

Add a one-line disclaimer in UI once (not on every card).

Respect the cooldown and per-set cap to avoid over-churn.

TL;DR

Flip betting_opps_enabled on, and the engine will (a) identify high-leverage pre-moments, (b) ask OpenAI for a 2-line novice explanation, and (c) emit a single TENNIS_AI_OPPORTUNITY alert with proper dedup/cooldown. Users can toggle it; admins can gate priority/cost—and you’ll get meaningful, easy-to-understand heads-ups before the flashy moments land.
